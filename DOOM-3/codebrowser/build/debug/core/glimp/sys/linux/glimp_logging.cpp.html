1537GLimp_EnableLogging

1540void GLimp_EnableLogging(bool enable) {
1541 static bool isEnabled = false;
1542 static idStr ospath;
1543 static int initialFrames;
1544
1545 // return if we're already active
1546 if ( isEnabled && enable ) {
1547 // decrement log counter and stop if it has reached 0
1548 r_logFile.SetInteger( r_logFile.GetInteger() - 1 );
1549 if ( r_logFile.GetInteger() ) {
1550 return;
1551 }
1552#if ID_LOG_TO_STDOUT
1553 common->Printf( "end stdout GL loggging after %i frames.\n", initialFrames );
1554#else
1555 common->Printf( "closing GL logfile '%s' after %i frames.\n", ospath.c_str(), initialFrames );
1556
1557 fclose( tr.logFile );
1558#endif
1559 enable = false;
1560 tr.logFile = NULL;
1561 }
1562
1563 // return if we're already disabled
1564 if ( !enable && !isEnabled ) {
1565 return;
1566 }
1567
1568 isEnabled = enable;
1569
1570 if ( enable ) {
1571 if ( !tr.logFile ) {
1572 struct tm *newtime;
1573 time_t aclock;
1574 idStr qpath;
1575 int i;
1576
1577 initialFrames = r_logFile.GetInteger();
1578
1579#if ID_LOG_TO_STDOUT
1580 tr.logFile = fdopen( STDOUT_FILENO, "w" );
1581#else
1582 // scan for an unused filename
1583 for ( i = 0 ; i < 9999 ; i++ ) {
1584 sprintf( qpath, "renderlog_%i.txt", i );
1585 if ( fileSystem->ReadFile( qpath, NULL, NULL ) == -1 ) {
1586 break; // use this name
1587 }
1588 }
1589
1590 ospath = fileSystem->RelativePathToOSPath( qpath );
1591 tr.logFile = fopen( ospath, "wt" );
1592#endif
1593
1594 // write the time out to the top of the file
1595 time( &aclock );
1596 newtime = localtime( &aclock );
1597 fprintf( tr.logFile, "// %s", asctime( newtime ) );
1598 fprintf( tr.logFile, "// %s\n\n", com_version.GetString() );
1599 }
1600
1601 GLimp_BindLogging();
1602 } else {
1603
1604 GLimp_BindNative();
1605 }

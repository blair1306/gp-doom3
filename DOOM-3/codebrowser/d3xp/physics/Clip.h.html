139ID_INLINE void idClipModel::Translate( const idVec3 &translation ) {
140 Unlink();
141 origin += translation;
142}
143
144ID_INLINE void idClipModel::Rotate( const idRotation &rotation ) {
145 Unlink();
146 origin *= rotation;
147 axis *= rotation.ToMat3();
148}

333ID_INLINE bool idClip::TracePoint( trace_t &results, const idVec3 &start, const idVec3 &end, int contentMask, const idEntity *passEntity ) {
334 Translation( results, start, end, NULL, mat3_identity, contentMask, passEntity );
335 return ( results.fraction < 1.0f );
336}
337
338ID_INLINE bool idClip::TraceBounds( trace_t &results, const idVec3 &start, const idVec3 &end, const idBounds &bounds, int contentMask, const idEntity *passEntity ) {
339 temporaryClipModel.LoadModel( idTraceModel( bounds ) );
340 Translation( results, start, end, &temporaryClipModel, mat3_identity, contentMask, passEntity );
341 return ( results.fraction < 1.0f );
342}

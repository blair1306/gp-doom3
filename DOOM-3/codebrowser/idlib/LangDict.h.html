29#ifndef __LANGDICT_H__
30#define __LANGDICT_H__
31
32/*
33===============================================================================
34
35 Simple dictionary specifically for the localized string tables.
36
37===============================================================================
38*/
39
40class idLangKeyValue {
41public:
42 idStr key;
43 idStr value;
44};
45
46class idLangDict {
47public:
48 idLangDict( void );
49 ~idLangDict( void );
50
51 void Clear( void );
52 bool Load( const char *fileName, bool clear = true );
53 void Save( const char *fileName );
54
55 const char * AddString( const char *str );
56 const char * GetString( const char *str ) const;
57
58 // adds the value and key as passed (doesn't generate a "#str_xxxxx" key or ensure the key/value pair is unique)
59 void AddKeyVal( const char *key, const char *val );
60
61 int GetNumKeyVals( void ) const;
62 const idLangKeyValue * GetKeyVal( int i ) const;
63
64 void SetBaseID(int id) { baseID = id; };
65
66private:
67 idList<idLangKeyValue> args;
68 idHashIndex hash;
69
70 bool ExcludeString( const char *str ) const;
71 int GetNextId( void ) const;
72 int GetHashKey( const char *str ) const;
73
74 int baseID;
75};
76
77#endif /* !__LANGDICT_H__ */

29#ifndef __BASE64_H__
30#define __BASE64_H__
31
32/*
33===============================================================================
34
35 base64
36
37===============================================================================
38*/
39
40class idBase64 {
41public:
42 idBase64( void );
43 idBase64( const idStr &s );
44 ~idBase64( void );
45
46 void Encode( const byte *from, int size );
47 void Encode( const idStr &src );
48 int DecodeLength( void ) const; // minimum size in bytes of destination buffer for decoding
49 int Decode( byte *to ) const; // does not append a \0 - needs a DecodeLength() bytes buffer
50 void Decode( idStr &dest ) const; // decodes the binary content to an idStr (a bit dodgy, \0 and other non-ascii are possible in the decoded content)
51 void Decode( idFile *dest ) const;
52
53 const char *c_str() const;
54
55 void operator=( const idStr &s );
56
57private:
58 byte * data;
59 int len;
60 int alloced;
61
62 void Init( void );
63 void Release( void );
64 void EnsureAlloced( int size );
65};
66
67ID_INLINE idBase64::idBase64( void ) {
68 Init();
69}
70
71ID_INLINE idBase64::idBase64( const idStr &s ) {
72 Init();
73 *this = s;
74}
75
76ID_INLINE idBase64::~idBase64( void ) {
77 Release();
78}
79
80ID_INLINE const char *idBase64::c_str( void ) const {
81 return (const char *)data;
82}
83
84ID_INLINE void idBase64::Init( void ) {
85 len = 0;
86 alloced = 0;
87 data = NULL;
88}
89
90ID_INLINE void idBase64::Release( void ) {
91 if ( data ) {
92 delete[] data;
93 }
94 Init();
95}
96
97ID_INLINE void idBase64::EnsureAlloced( int size ) {
98 if ( size > alloced ) {
99 Release();
100 }
101 data = new byte[size];
102 alloced = size;
103}
104
105ID_INLINE void idBase64::operator=( const idStr &s ) {
106 EnsureAlloced( s.Length()+1 ); // trailing \0 - beware, this does a Release
107 strcpy( (char *)data, s.c_str() );
108 len = s.Length();
109}

111#endif /* !__BASE64_H__ */

358ID_INLINE idStr::idStr( const idStr &text, int start, int end ) {
359 int i;
360 int l;
361
362 Init();
363 if ( end > text.Length() ) {
364 end = text.Length();
365 }
366 if ( start > text.Length() ) {
367 start = text.Length();
368 } else if ( start < 0 ) {
369 start = 0;
370 }
371
372 l = end - start;
373 if ( l < 0 ) {
374 l = 0;
375 }
376
377 EnsureAlloced( l + 1 );
378
379 for ( i = 0; i < l; i++ ) {
380 data[ i ] = text[ start + i ];
381 }
382
383 data[ l ] = '\0';
384 len = l;
385}

466ID_INLINE idStr::idStr( const float f ) {
467 char text[ 64 ];
468 int l;
469
470 Init();
471 l = idStr::snPrintf( text, sizeof( text ), "%f", f );
472 while( l > 0 && text[l-1] == '0' ) text[--l] = '\0';
473 while( l > 0 && text[l-1] == '.' ) text[--l] = '\0';
474 EnsureAlloced( l + 1 );
475 strcpy( data, text );
476 len = l;
477}

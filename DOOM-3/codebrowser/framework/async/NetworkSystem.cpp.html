42*/
43void idNetworkSystem::ServerSendReliableMessage( int clientNum, const idBitMsg &msg ) {
44 if ( idAsyncNetwork::server.IsActive() ) {
45 idAsyncNetwork::server.SendReliableGameMessage( clientNum, msg );

147==================
148*/
149void idNetworkSystem::ClientSendReliableMessage( const idBitMsg &msg ) {
150 if ( idAsyncNetwork::client.IsActive() ) {
151 idAsyncNetwork::client.SendReliableGameMessage( msg );
152 } else if ( idAsyncNetwork::server.IsActive() ) {
153 idAsyncNetwork::server.LocalClientSendReliableMessage( msg );
154 }

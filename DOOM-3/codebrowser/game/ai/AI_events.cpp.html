317=====================
318idAI::Event_FindEnemy
319=====================
320*/
321void idAI::Event_FindEnemy( int useFOV ) {
322 int i;
323 idEntity *ent;
324 idActor *actor;
325
326 if ( gameLocal.InPlayerPVS( this ) ) {
327 for ( i = 0; i < gameLocal.numClients ; i++ ) {
328 ent = gameLocal.entities[ i ];
329
330 if ( !ent || !ent->IsType( idActor::Type ) ) {
331 continue;
332 }
333
334 actor = static_cast<idActor *>( ent );
335 if ( ( actor->health <= 0 ) || !( ReactionTo( actor ) & ATTACK_ON_SIGHT ) ) {
336 continue;
337 }
338
339 if ( CanSee( actor, useFOV != 0 ) ) {
340 idThread::ReturnEntity( actor );
341 return;
342 }
343 }
344 }
345
346 idThread::ReturnEntity( NULL );
347}

349/*
350=====================
351idAI::Event_FindEnemyAI
352=====================
353*/
354void idAI::Event_FindEnemyAI( int useFOV ) {
355 idEntity *ent;
356 idActor *actor;
357 idActor *bestEnemy;
358 float bestDist;
359 float dist;
360 idVec3 delta;
361 pvsHandle_t pvs;
362
363 pvs = gameLocal.pvs.SetupCurrentPVS( GetPVSAreas(), GetNumPVSAreas() );
364
365 bestDist = idMath::INFINITY;
366 bestEnemy = NULL;
367 for ( ent = gameLocal.activeEntities.Next(); ent != NULL; ent = ent->activeNode.Next() ) {
368 if ( ent->fl.hidden || ent->fl.isDormant || !ent->IsType( idActor::Type ) ) {
369 continue;
370 }
371
372 actor = static_cast<idActor *>( ent );
373 if ( ( actor->health <= 0 ) || !( ReactionTo( actor ) & ATTACK_ON_SIGHT ) ) {
374 continue;
375 }
376
377 if ( !gameLocal.pvs.InCurrentPVS( pvs, actor->GetPVSAreas(), actor->GetNumPVSAreas() ) ) {
378 continue;
379 }
380
381 delta = physicsObj.GetOrigin() - actor->GetPhysics()->GetOrigin();
382 dist = delta.LengthSqr();
383 if ( ( dist < bestDist ) && CanSee( actor, useFOV != 0 ) ) {
384 bestDist = dist;
385 bestEnemy = actor;
386 }
387 }
388
389 gameLocal.pvs.FreeCurrentPVS( pvs );
390 idThread::ReturnEntity( bestEnemy );
391}

393/*
394=====================
395idAI::Event_FindEnemyInCombatNodes
396=====================
397*/
398void idAI::Event_FindEnemyInCombatNodes( void ) {
399 int i, j;
400 idCombatNode *node;
401 idEntity *ent;
402 idEntity *targetEnt;
403 idActor *actor;
404
405 if ( !gameLocal.InPlayerPVS( this ) ) {
406 // don't locate the player when we're not in his PVS
407 idThread::ReturnEntity( NULL );
408 return;
409 }
410
411 for ( i = 0; i < gameLocal.numClients ; i++ ) {
412 ent = gameLocal.entities[ i ];
413
414 if ( !ent || !ent->IsType( idActor::Type ) ) {
415 continue;
416 }
417
418 actor = static_cast<idActor *>( ent );
419 if ( ( actor->health <= 0 ) || !( ReactionTo( actor ) & ATTACK_ON_SIGHT ) ) {
420 continue;
421 }
422
423 for( j = 0; j < targets.Num(); j++ ) {
424 targetEnt = targets[ j ].GetEntity();
425 if ( !targetEnt || !targetEnt->IsType( idCombatNode::Type ) ) {
426 continue;
427 }
428
429 node = static_cast<idCombatNode *>( targetEnt );
430 if ( !node->IsDisabled() && node->EntityInView( actor, actor->GetPhysics()->GetOrigin() ) ) {
431 idThread::ReturnEntity( actor );
432 return;
433 }
434 }
435 }
436
437 idThread::ReturnEntity( NULL );
438}

440/*
441=====================
442idAI::Event_ClosestReachableEnemyOfEntity
443=====================
444*/
445void idAI::Event_ClosestReachableEnemyOfEntity( idEntity *team_mate ) {
446 idActor *actor;
447 idActor *ent;
448 idActor *bestEnt;
449 float bestDistSquared;
450 float distSquared;
451 idVec3 delta;
452 int areaNum;
453 int enemyAreaNum;
454 aasPath_t path;
455
456 if ( !team_mate->IsType( idActor::Type ) ) {
457 gameLocal.Error( "Entity '%s' is not an AI character or player", team_mate->GetName() );
458 }
459
460 actor = static_cast<idActor *>( team_mate );
461
462 const idVec3 &origin = physicsObj.GetOrigin();
463 areaNum = PointReachableAreaNum( origin );
464
465 bestDistSquared = idMath::INFINITY;
466 bestEnt = NULL;
467 for( ent = actor->enemyList.Next(); ent != NULL; ent = ent->enemyNode.Next() ) {
468 if ( ent->fl.hidden ) {
469 continue;
470 }
471 delta = ent->GetPhysics()->GetOrigin() - origin;
472 distSquared = delta.LengthSqr();
473 if ( distSquared < bestDistSquared ) {
474 const idVec3 &enemyPos = ent->GetPhysics()->GetOrigin();
475 enemyAreaNum = PointReachableAreaNum( enemyPos );
476 if ( ( areaNum != 0 ) && PathToGoal( path, areaNum, origin, enemyAreaNum, enemyPos ) ) {
477 bestEnt = ent;
478 bestDistSquared = distSquared;
479 }
480 }
481 }
482
483 idThread::ReturnEntity( bestEnt );
484}

486/*
487=====================
488idAI::Event_HeardSound
489=====================
490*/
491void idAI::Event_HeardSound( int ignore_team ) {
492 // check if we heard any sounds in the last frame
493 idActor *actor = gameLocal.GetAlertEntity();
494 if ( actor && ( !ignore_team || ( ReactionTo( actor ) & ATTACK_ON_SIGHT ) ) && gameLocal.InPlayerPVS( this ) ) {
495 idVec3 pos = actor->GetPhysics()->GetOrigin();
496 idVec3 org = physicsObj.GetOrigin();
497 float dist = ( pos - org ).LengthSqr();
498 if ( dist < Square( AI_HEARING_RANGE ) ) {

600/*
601=====================
602idAI::Event_LaunchMissile
603=====================
604*/
605void idAI::Event_LaunchMissile( const idVec3 &org, const idAngles &ang ) {
606 idVec3 start;
607 trace_t tr;
608 idBounds projBounds;
609 const idClipModel *projClip;
610 idMat3 axis;
611 float distance;
612
613 if ( !projectileDef ) {
614 gameLocal.Warning( "%s (%s) doesn't have a projectile specified", name.c_str(), GetEntityDefName() );
615 idThread::ReturnEntity( NULL );
616 return;
617 }
618
619 axis = ang.ToMat3();
620 if ( !projectile.GetEntity() ) {
621 CreateProjectile( org, axis[ 0 ] );
622 }
623
624 // make sure the projectile starts inside the monster bounding box
625 const idBounds &ownerBounds = physicsObj.GetAbsBounds();
626 projClip = projectile.GetEntity()->GetPhysics()->GetClipModel();
627 projBounds = projClip->GetBounds().Rotate( projClip->GetAxis() );
628
629 // check if the owner bounds is bigger than the projectile bounds
630 if ( ( ( ownerBounds[1][0] - ownerBounds[0][0] ) > ( projBounds[1][0] - projBounds[0][0] ) ) &&
631 ( ( ownerBounds[1][1] - ownerBounds[0][1] ) > ( projBounds[1][1] - projBounds[0][1] ) ) &&
632 ( ( ownerBounds[1][2] - ownerBounds[0][2] ) > ( projBounds[1][2] - projBounds[0][2] ) ) ) {
633 if ( (ownerBounds - projBounds).RayIntersection( org, viewAxis[ 0 ], distance ) ) {
634 start = org + distance * viewAxis[ 0 ];
635 } else {
636 start = ownerBounds.GetCenter();
637 }
638 } else {
639 // projectile bounds bigger than the owner bounds, so just start it from the center
640 start = ownerBounds.GetCenter();
641 }
642
643 gameLocal.clip.Translation( tr, start, org, projClip, projClip->GetAxis(), MASK_SHOT_RENDERMODEL, this );
644
645 // launch the projectile
646 idThread::ReturnEntity( projectile.GetEntity() );
647 projectile.GetEntity()->Launch( tr.endpos, axis[ 0 ], vec3_origin );
648 projectile = NULL;
649
650 TriggerWeaponEffects( tr.endpos );
651
652 lastAttackTime = gameLocal.time;
653}

1099/*
1100=====================
1101idAI::Event_GetCombatNode
1102=====================
1103*/
1104void idAI::Event_GetCombatNode( void ) {
1105 int i;
1106 float dist;
1107 idEntity *targetEnt;
1108 idCombatNode *node;
1109 float bestDist;
1110 idCombatNode *bestNode;
1111 idActor *enemyEnt = enemy.GetEntity();
1112
1113 if ( !targets.Num() ) {
1114 // no combat nodes
1115 idThread::ReturnEntity( NULL );
1116 return;
1117 }
1118
1119 if ( !enemyEnt || !EnemyPositionValid() ) {
1120 // don't return a combat node if we don't have an enemy or
1121 // if we can see he's not in the last place we saw him
1122 idThread::ReturnEntity( NULL );
1123 return;
1124 }
1125
1126 // find the closest attack node that can see our enemy and is closer than our enemy
1127 bestNode = NULL;
1128 const idVec3 &myPos = physicsObj.GetOrigin();
1129 bestDist = ( myPos - lastVisibleEnemyPos ).LengthSqr();
1130 for( i = 0; i < targets.Num(); i++ ) {
1131 targetEnt = targets[ i ].GetEntity();
1132 if ( !targetEnt || !targetEnt->IsType( idCombatNode::Type ) ) {
1133 continue;
1134 }
1135
1136 node = static_cast<idCombatNode *>( targetEnt );
1137 if ( !node->IsDisabled() && node->EntityInView( enemyEnt, lastVisibleEnemyPos ) ) {
1138 idVec3 org = node->GetPhysics()->GetOrigin();
1139 dist = ( myPos - org ).LengthSqr();
1140 if ( dist < bestDist ) {
1141 bestNode = node;
1142 bestDist = dist;
1143 }
1144 }
1145 }
1146
1147 idThread::ReturnEntity( bestNode );
1148}

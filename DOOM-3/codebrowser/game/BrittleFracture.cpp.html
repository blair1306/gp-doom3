79idBrittleFracture::~idBrittleFracture( void ) {

81

92/*
93================
94idBrittleFracture::Save
95================
96*/
97void idBrittleFracture::Save( idSaveGame *savefile ) const {
98 int i, j;
99
100 savefile->WriteInt( health );
101 entityFlags_s flags = fl;
102 LittleBitField( &flags, sizeof( flags ) );
103 savefile->Write( &flags, sizeof( flags ) );
104
105 // setttings
106 savefile->WriteMaterial( material );
107 savefile->WriteMaterial( decalMaterial );
108 savefile->WriteFloat( decalSize );
109 savefile->WriteFloat( maxShardArea );
110 savefile->WriteFloat( maxShatterRadius );
111 savefile->WriteFloat( minShatterRadius );
112 savefile->WriteFloat( linearVelocityScale );
113 savefile->WriteFloat( angularVelocityScale );
114 savefile->WriteFloat( shardMass );
115 savefile->WriteFloat( density );
116 savefile->WriteFloat( friction );
117 savefile->WriteFloat( bouncyness );
118 savefile->WriteString( fxFracture );
119
120 // state
121 savefile->WriteBounds( bounds );
122 savefile->WriteBool( disableFracture );
123
124 savefile->WriteInt( lastRenderEntityUpdate );
125 savefile->WriteBool( changed );
126
127 savefile->WriteStaticObject( physicsObj );
128
129 savefile->WriteInt( shards.Num() );
130 for ( i = 0; i < shards.Num(); i++ ) {
131 savefile->WriteWinding( shards[i]->winding );
132
133 savefile->WriteInt( shards[i]->decals.Num() );
134 for ( j = 0; j < shards[i]->decals.Num(); j++ ) {
135 savefile->WriteWinding( *shards[i]->decals[j] );
136 }
137
138 savefile->WriteInt( shards[i]->neighbours.Num() );
139 for ( j = 0; j < shards[i]->neighbours.Num(); j++ ) {
140 int index = shards.FindIndex(shards[i]->neighbours[j]);
141 assert(index != -1);
142 savefile->WriteInt( index );
143 }
144
145 savefile->WriteInt( shards[i]->edgeHasNeighbour.Num() );
146 for ( j = 0; j < shards[i]->edgeHasNeighbour.Num(); j++ ) {
147 savefile->WriteBool( shards[i]->edgeHasNeighbour[j] );
148 }
149
150 savefile->WriteInt( shards[i]->droppedTime );
151 savefile->WriteInt( shards[i]->islandNum );
152 savefile->WriteBool( shards[i]->atEdge );
153 savefile->WriteStaticObject( shards[i]->physicsObj );
154 }
155}

159idBrittleFracture::Restore

162void idBrittleFracture::Restore( idRestoreGame *savefile ) {
163 int i, j , num;
164
165 renderEntity.hModel = renderModelManager->AllocModel();
166 renderEntity.hModel->InitEmpty( brittleFracture_SnapshotName );
167 renderEntity.callback = idBrittleFracture::ModelCallback;
168 renderEntity.noShadow = true;
169 renderEntity.noSelfShadow = true;
170 renderEntity.noDynamicInteractions = false;
171
172 savefile->ReadInt( health );
173 savefile->Read( &fl, sizeof( fl ) );
174 LittleBitField( &fl, sizeof( fl ) );
175
176 // setttings
177 savefile->ReadMaterial( material );
178 savefile->ReadMaterial( decalMaterial );
179 savefile->ReadFloat( decalSize );
180 savefile->ReadFloat( maxShardArea );
181 savefile->ReadFloat( maxShatterRadius );
182 savefile->ReadFloat( minShatterRadius );
183 savefile->ReadFloat( linearVelocityScale );
184 savefile->ReadFloat( angularVelocityScale );
185 savefile->ReadFloat( shardMass );
186 savefile->ReadFloat( density );
187 savefile->ReadFloat( friction );
188 savefile->ReadFloat( bouncyness );
189 savefile->ReadString( fxFracture );
190
191 // state
192 savefile->ReadBounds(bounds);
193 savefile->ReadBool( disableFracture );
194
195 savefile->ReadInt( lastRenderEntityUpdate );
196 savefile->ReadBool( changed );
197
198 savefile->ReadStaticObject( physicsObj );
199 RestorePhysics( &physicsObj );
200
201 savefile->ReadInt( num );
202 shards.SetNum( num );
203 for ( i = 0; i < num; i++ ) {
204 shards[i] = new shard_t;
205 }
206
207 for ( i = 0; i < num; i++ ) {
208 savefile->ReadWinding( shards[i]->winding );
209
210 savefile->ReadInt( j );
211 shards[i]->decals.SetNum( j );
212 for ( j = 0; j < shards[i]->decals.Num(); j++ ) {
213 shards[i]->decals[j] = new idFixedWinding;
214 savefile->ReadWinding( *shards[i]->decals[j] );
215 }
216
217 savefile->ReadInt( j );
218 shards[i]->neighbours.SetNum( j );
219 for ( j = 0; j < shards[i]->neighbours.Num(); j++ ) {
220 int index;
221 savefile->ReadInt( index );
222 assert(index != -1);
223 shards[i]->neighbours[j] = shards[index];
224 }
225
226 savefile->ReadInt( j );
227 shards[i]->edgeHasNeighbour.SetNum( j );
228 for ( j = 0; j < shards[i]->edgeHasNeighbour.Num(); j++ ) {
229 savefile->ReadBool( shards[i]->edgeHasNeighbour[j] );
230 }
231
232 savefile->ReadInt( shards[i]->droppedTime );
233 savefile->ReadInt( shards[i]->islandNum );
234 savefile->ReadBool( shards[i]->atEdge );
235 savefile->ReadStaticObject( shards[i]->physicsObj );
236 if ( shards[i]->droppedTime < 0 ) {
237 shards[i]->clipModel = physicsObj.GetClipModel( i );
238 } else {
239 shards[i]->clipModel = shards[i]->physicsObj.GetClipModel();
240 }
241 }

245================
246idBrittleFracture::Spawn
247================
248*/
249void idBrittleFracture::Spawn( void ) {
250
251 // get shard properties
252 decalMaterial = declManager->FindMaterial( spawnArgs.GetString( "mtr_decal" ) );
253 decalSize = spawnArgs.GetFloat( "decalSize", "40" );
254 maxShardArea = spawnArgs.GetFloat( "maxShardArea", "200" );
255 maxShardArea = idMath::ClampFloat( 100, 10000, maxShardArea );
256 maxShatterRadius = spawnArgs.GetFloat( "maxShatterRadius", "40" );
257 minShatterRadius = spawnArgs.GetFloat( "minShatterRadius", "10" );
258 linearVelocityScale = spawnArgs.GetFloat( "linearVelocityScale", "0.1" );
259 angularVelocityScale = spawnArgs.GetFloat( "angularVelocityScale", "40" );
260 fxFracture = spawnArgs.GetString( "fx" );
261
262 // get rigid body properties
263 shardMass = spawnArgs.GetFloat( "shardMass", "20" );
264 shardMass = idMath::ClampFloat( 0.001f, 1000.0f, shardMass );
265 spawnArgs.GetFloat( "density", "0.1", density );
266 density = idMath::ClampFloat( 0.001f, 1000.0f, density );
267 spawnArgs.GetFloat( "friction", "0.4", friction );
268 friction = idMath::ClampFloat( 0.0f, 1.0f, friction );
269 spawnArgs.GetFloat( "bouncyness", "0.01", bouncyness );
270 bouncyness = idMath::ClampFloat( 0.0f, 1.0f, bouncyness );
271
272 disableFracture = spawnArgs.GetBool( "disableFracture", "0" );
273 health = spawnArgs.GetInt( "health", "40" );
274 fl.takedamage = true;
275
276 // FIXME: set "bleed" so idProjectile calls AddDamageEffect
277 spawnArgs.SetBool( "bleed", 1 );
278
279 CreateFractures( renderEntity.hModel );
280
281 FindNeighbours();
282
283 renderEntity.hModel = renderModelManager->AllocModel();
284 renderEntity.hModel->InitEmpty( brittleFracture_SnapshotName );
285 renderEntity.callback = idBrittleFracture::ModelCallback;
286 renderEntity.noShadow = true;
287 renderEntity.noSelfShadow = true;
288 renderEntity.noDynamicInteractions = false;
289}

325/*
326================
327idBrittleFracture::UpdateRenderEntity
328================
329*/
330bool idBrittleFracture::UpdateRenderEntity( renderEntity_s *renderEntity, const renderView_t *renderView ) const {
331 int i, j, k, n, msec, numTris, numDecalTris;
332 float fade;
333 dword packedColor;
334 srfTriangles_t *tris, *decalTris;
335 modelSurface_t surface;
336 idDrawVert *v;
337 idPlane plane;
338 idMat3 tangents;
339
340 // this may be triggered by a model trace or other non-view related source,
341 // to which we should look like an empty model
342 if ( !renderView ) {
343 return false;
344 }
345
346 // don't regenerate it if it is current
347 if ( lastRenderEntityUpdate == gameLocal.time || !changed ) {
348 return false;
349 }
350
351 lastRenderEntityUpdate = gameLocal.time;
352 changed = false;
353
354 numTris = 0;
355 numDecalTris = 0;
356 for ( i = 0; i < shards.Num(); i++ ) {
357 n = shards[i]->winding.GetNumPoints();
358 if ( n > 2 ) {
359 numTris += n - 2;
360 }
361 for ( k = 0; k < shards[i]->decals.Num(); k++ ) {
362 n = shards[i]->decals[k]->GetNumPoints();
363 if ( n > 2 ) {
364 numDecalTris += n - 2;
365 }
366 }
367 }
368
369 // FIXME: re-use model surfaces
370 renderEntity->hModel->InitEmpty( brittleFracture_SnapshotName );
371
372 // allocate triangle surfaces for the fractures and decals
373 tris = renderEntity->hModel->AllocSurfaceTriangles( numTris * 3, material->ShouldCreateBackSides() ? numTris * 6 : numTris * 3 );
374 decalTris = renderEntity->hModel->AllocSurfaceTriangles( numDecalTris * 3, decalMaterial->ShouldCreateBackSides() ? numDecalTris * 6 : numDecalTris * 3 );
375
376 for ( i = 0; i < shards.Num(); i++ ) {
377 const idVec3 &origin = shards[i]->clipModel->GetOrigin();
378 const idMat3 &axis = shards[i]->clipModel->GetAxis();
379
380 fade = 1.0f;
381 if ( shards[i]->droppedTime >= 0 ) {
382 msec = gameLocal.time - shards[i]->droppedTime - SHARD_FADE_START;
383 if ( msec > 0 ) {
384 fade = 1.0f - (float) msec / ( SHARD_ALIVE_TIME - SHARD_FADE_START );
385 }
386 }
387 packedColor = PackColor( idVec4( renderEntity->shaderParms[ SHADERPARM_RED ] * fade,
388 renderEntity->shaderParms[ SHADERPARM_GREEN ] * fade,
389 renderEntity->shaderParms[ SHADERPARM_BLUE ] * fade,
390 fade ) );
391
392 const idWinding &winding = shards[i]->winding;
393
394 winding.GetPlane( plane );
395 tangents = ( plane.Normal() * axis ).ToMat3();
396
397 for ( j = 2; j < winding.GetNumPoints(); j++ ) {
398
399 v = &tris->verts[tris->numVerts++];
400 v->Clear();
401 v->xyz = origin + winding[0].ToVec3() * axis;
402 v->st[0] = winding[0].s;
403 v->st[1] = winding[0].t;
404 v->normal = tangents[0];
405 v->tangents[0] = tangents[1];
406 v->tangents[1] = tangents[2];
407 v->SetColor( packedColor );
408
409 v = &tris->verts[tris->numVerts++];
410 v->Clear();
411 v->xyz = origin + winding[j-1].ToVec3() * axis;
412 v->st[0] = winding[j-1].s;
413 v->st[1] = winding[j-1].t;
414 v->normal = tangents[0];
415 v->tangents[0] = tangents[1];
416 v->tangents[1] = tangents[2];
417 v->SetColor( packedColor );
418
419 v = &tris->verts[tris->numVerts++];
420 v->Clear();
421 v->xyz = origin + winding[j].ToVec3() * axis;
422 v->st[0] = winding[j].s;
423 v->st[1] = winding[j].t;
424 v->normal = tangents[0];
425 v->tangents[0] = tangents[1];
426 v->tangents[1] = tangents[2];
427 v->SetColor( packedColor );
428
429 tris->indexes[tris->numIndexes++] = tris->numVerts - 3;
430 tris->indexes[tris->numIndexes++] = tris->numVerts - 2;
431 tris->indexes[tris->numIndexes++] = tris->numVerts - 1;
432
433 if ( material->ShouldCreateBackSides() ) {
434
435 tris->indexes[tris->numIndexes++] = tris->numVerts - 2;
436 tris->indexes[tris->numIndexes++] = tris->numVerts - 3;
437 tris->indexes[tris->numIndexes++] = tris->numVerts - 1;
438 }
439 }
440
441 for ( k = 0; k < shards[i]->decals.Num(); k++ ) {
442 const idWinding &decalWinding = *shards[i]->decals[k];
443
444 for ( j = 2; j < decalWinding.GetNumPoints(); j++ ) {
445
446 v = &decalTris->verts[decalTris->numVerts++];
447 v->Clear();
448 v->xyz = origin + decalWinding[0].ToVec3() * axis;
449 v->st[0] = decalWinding[0].s;
450 v->st[1] = decalWinding[0].t;
451 v->normal = tangents[0];
452 v->tangents[0] = tangents[1];
453 v->tangents[1] = tangents[2];
454 v->SetColor( packedColor );
455
456 v = &decalTris->verts[decalTris->numVerts++];
457 v->Clear();
458 v->xyz = origin + decalWinding[j-1].ToVec3() * axis;
459 v->st[0] = decalWinding[j-1].s;
460 v->st[1] = decalWinding[j-1].t;
461 v->normal = tangents[0];
462 v->tangents[0] = tangents[1];
463 v->tangents[1] = tangents[2];
464 v->SetColor( packedColor );
465
466 v = &decalTris->verts[decalTris->numVerts++];
467 v->Clear();
468 v->xyz = origin + decalWinding[j].ToVec3() * axis;
469 v->st[0] = decalWinding[j].s;
470 v->st[1] = decalWinding[j].t;
471 v->normal = tangents[0];
472 v->tangents[0] = tangents[1];
473 v->tangents[1] = tangents[2];
474 v->SetColor( packedColor );
475
476 decalTris->indexes[decalTris->numIndexes++] = decalTris->numVerts - 3;
477 decalTris->indexes[decalTris->numIndexes++] = decalTris->numVerts - 2;
478 decalTris->indexes[decalTris->numIndexes++] = decalTris->numVerts - 1;
479
480 if ( decalMaterial->ShouldCreateBackSides() ) {
481
482 decalTris->indexes[decalTris->numIndexes++] = decalTris->numVerts - 2;
483 decalTris->indexes[decalTris->numIndexes++] = decalTris->numVerts - 3;
484 decalTris->indexes[decalTris->numIndexes++] = decalTris->numVerts - 1;
485 }
486 }
487 }
488 }
489
490 tris->tangentsCalculated = true;
491 decalTris->tangentsCalculated = true;
492
493 SIMDProcessor->MinMax( tris->bounds[0], tris->bounds[1], tris->verts, tris->numVerts );
494 SIMDProcessor->MinMax( decalTris->bounds[0], decalTris->bounds[1], decalTris->verts, decalTris->numVerts );
495
496 memset( &surface, 0, sizeof( surface ) );
497 surface.shader = material;
498 surface.id = 0;
499 surface.geometry = tris;
500 renderEntity->hModel->AddSurface( surface );
501
502 memset( &surface, 0, sizeof( surface ) );
503 surface.shader = decalMaterial;
504 surface.id = 1;
505 surface.geometry = decalTris;
506 renderEntity->hModel->AddSurface( surface );
507
508 return true;
509}

532void idBrittleFracture::Present() {
533
534 // don't present to the renderer if the entity hasn't changed
535 if ( !( thinkFlags & TH_UPDATEVISUALS ) ) {
536 return;
537 }
538 BecomeInactive( TH_UPDATEVISUALS );
539
540 renderEntity.bounds = bounds;
541 renderEntity.origin.Zero();
542 renderEntity.axis.Identity();
543

545 renderEntity.forceUpdate = true;
546

550 } else {
551 gameRenderWorld->UpdateEntityDef( modelDefHandle, &renderEntity );
552 }
553
554 changed = true;
555}

557/*
558================
559idBrittleFracture::Think
560================
561*/
562void idBrittleFracture::Think( void ) {
563 int i, startTime, endTime, droppedTime;
564 shard_t *shard;
565 bool atRest = true, fading = false;
566
567 // remove overdue shards
568 for ( i = 0; i < shards.Num(); i++ ) {
569 droppedTime = shards[i]->droppedTime;
570 if ( droppedTime != -1 ) {
571 if ( gameLocal.time - droppedTime > SHARD_ALIVE_TIME ) {
572 RemoveShard( i );
573 i--;
574 }
575 fading = true;
576 }
577 }
578
579 // remove the entity when nothing is visible
580 if ( !shards.Num() ) {
581 PostEventMS( &EV_Remove, 0 );
582 return;
583 }
584
585 if ( thinkFlags & TH_PHYSICS ) {
586
587 startTime = gameLocal.previousTime;
588 endTime = gameLocal.time;
589
590 // run physics on shards
591 for ( i = 0; i < shards.Num(); i++ ) {
592 shard = shards[i];
593
594 if ( shard->droppedTime == -1 ) {
595 continue;
596 }
597
598 shard->physicsObj.Evaluate( endTime - startTime, endTime );
599
600 if ( !shard->physicsObj.IsAtRest() ) {
601 atRest = false;
602 }
603 }
604
605 if ( atRest ) {
606 BecomeInactive( TH_PHYSICS );
607 } else {
608 BecomeActive( TH_PHYSICS );
609 }
610 }
611
612 if ( !atRest || bounds.IsCleared() ) {
613 bounds.Clear();
614 for ( i = 0; i < shards.Num(); i++ ) {
615 bounds.AddBounds( shards[i]->clipModel->GetAbsBounds() );
616 }
617 }
618
619 if ( fading ) {
620 BecomeActive( TH_UPDATEVISUALS | TH_THINK );
621 } else {
622 BecomeInactive( TH_THINK );
623 }
624
625 RunPhysics();
626 Present();
627}

665/*
666================
667idBrittleFracture::ProjectDecal
668================
669*/
670void idBrittleFracture::ProjectDecal( const idVec3 &point, const idVec3 &dir, const int time, const char *damageDefName ) {
671 int i, j, bits, clipBits;
672 float a, c, s;
673 idVec2 st[MAX_POINTS_ON_WINDING];
674 idVec3 origin;
675 idMat3 axis, axistemp;
676 idPlane textureAxis[2];
677
678 if ( gameLocal.isServer ) {
679 idBitMsg msg;
680 byte msgBuf[MAX_EVENT_PARAM_SIZE];
681
682 msg.Init( msgBuf, sizeof( msgBuf ) );
683 msg.BeginWriting();
684 msg.WriteFloat( point[0] );
685 msg.WriteFloat( point[1] );
686 msg.WriteFloat( point[2] );
687 msg.WriteFloat( dir[0] );
688 msg.WriteFloat( dir[1] );
689 msg.WriteFloat( dir[2] );
690 ServerSendEvent( EVENT_PROJECT_DECAL, &msg, true, -1 );
691 }
692
693 if ( time >= gameLocal.time ) {
694 // try to get the sound from the damage def
695 const idDeclEntityDef *damageDef = NULL;
696 const idSoundShader *sndShader = NULL;
697 if ( damageDefName ) {
698 damageDef = gameLocal.FindEntityDef( damageDefName, false );
699 if ( damageDef ) {
700 sndShader = declManager->FindSound( damageDef->dict.GetString( "snd_shatter", "" ) );
701 }
702 }
703
704 if ( sndShader ) {
705 StartSoundShader( sndShader, SND_CHANNEL_ANY, 0, false, NULL );
706 } else {
707 StartSound( "snd_bullethole", SND_CHANNEL_ANY, 0, false, NULL );
708 }
709 }
710
711 a = gameLocal.random.RandomFloat() * idMath::TWO_PI;
712 c = cos( a );
713 s = -sin( a );
714
715 axis[2] = -dir;
716 axis[2].Normalize();
717 axis[2].NormalVectors( axistemp[0], axistemp[1] );
718 axis[0] = axistemp[ 0 ] * c + axistemp[ 1 ] * s;
719 axis[1] = axistemp[ 0 ] * s + axistemp[ 1 ] * -c;
720
721 textureAxis[0] = axis[0] * ( 1.0f / decalSize );
722 textureAxis[0][3] = -( point * textureAxis[0].Normal() ) + 0.5f;
723
724 textureAxis[1] = axis[1] * ( 1.0f / decalSize );
725 textureAxis[1][3] = -( point * textureAxis[1].Normal() ) + 0.5f;
726
727 for ( i = 0; i < shards.Num(); i++ ) {
728 idFixedWinding &winding = shards[i]->winding;
729 origin = shards[i]->clipModel->GetOrigin();
730 axis = shards[i]->clipModel->GetAxis();
731 float d0, d1;
732
733 clipBits = -1;
734 for ( j = 0; j < winding.GetNumPoints(); j++ ) {
735 idVec3 p = origin + winding[j].ToVec3() * axis;
736
737 st[j].x = d0 = textureAxis[0].Distance( p );
738 st[j].y = d1 = textureAxis[1].Distance( p );
739
740 bits = FLOATSIGNBITSET( d0 );
741 d0 = 1.0f - d0;
742 bits |= FLOATSIGNBITSET( d1 ) << 2;
743 d1 = 1.0f - d1;
744 bits |= FLOATSIGNBITSET( d0 ) << 1;
745 bits |= FLOATSIGNBITSET( d1 ) << 3;
746
747 clipBits &= bits;
748 }
749
750 if ( clipBits ) {
751 continue;
752 }
753
754 idFixedWinding *decal = new idFixedWinding;
755 shards[i]->decals.Append( decal );
756
757 decal->SetNumPoints( winding.GetNumPoints() );
758 for ( j = 0; j < winding.GetNumPoints(); j++ ) {
759 (*decal)[j].ToVec3() = winding[j].ToVec3();
760 (*decal)[j].s = st[j].x;
761 (*decal)[j].t = st[j].y;
762 }
763 }
764
765 BecomeActive( TH_UPDATEVISUALS );
766}

768/*
769================
770idBrittleFracture::DropShard
771================
772*/
773void idBrittleFracture::DropShard( shard_t *shard, const idVec3 &point, const idVec3 &dir, const float impulse, const int time ) {
774 int i, j, clipModelId;
775 float dist, f;
776 idVec3 dir2, origin;
777 idMat3 axis;
778 shard_t *neighbour;
779
780 // don't display decals on dropped shards
781 shard->decals.DeleteContents( true );
782
783 // remove neighbour pointers of neighbours pointing to this shard
784 for ( i = 0; i < shard->neighbours.Num(); i++ ) {
785 neighbour = shard->neighbours[i];
786 for ( j = 0; j < neighbour->neighbours.Num(); j++ ) {
787 if ( neighbour->neighbours[j] == shard ) {
788 neighbour->neighbours.RemoveIndex( j );
789 break;
790 }
791 }
792 }
793
794 // remove neighbour pointers
795 shard->neighbours.Clear();
796
797 // remove the clip model from the static physics object
798 clipModelId = shard->clipModel->GetId();
799 physicsObj.SetClipModel( NULL, 1.0f, clipModelId, false );
800
801 origin = shard->clipModel->GetOrigin();
802 axis = shard->clipModel->GetAxis();
803
804 // set the dropped time for fading
805 shard->droppedTime = time;
806
807 dir2 = origin - point;
808 dist = dir2.Normalize();
809 f = dist > maxShatterRadius ? 1.0f : idMath::Sqrt( dist - minShatterRadius ) * ( 1.0f / idMath::Sqrt( maxShatterRadius - minShatterRadius ) );
810
811 // setup the physics
812 shard->physicsObj.SetSelf( this );
813 shard->physicsObj.SetClipModel( shard->clipModel, density );
814 shard->physicsObj.SetMass( shardMass );
815 shard->physicsObj.SetOrigin( origin );
816 shard->physicsObj.SetAxis( axis );
817 shard->physicsObj.SetBouncyness( bouncyness );
818 shard->physicsObj.SetFriction( 0.6f, 0.6f, friction );
819 shard->physicsObj.SetGravity( gameLocal.GetGravity() );
820 shard->physicsObj.SetContents( CONTENTS_RENDERMODEL );
821 shard->physicsObj.SetClipMask( MASK_SOLID | CONTENTS_MOVEABLECLIP );
822 shard->physicsObj.ApplyImpulse( 0, origin, impulse * linearVelocityScale * dir );
823 shard->physicsObj.SetAngularVelocity( dir.Cross( dir2 ) * ( f * angularVelocityScale ) );
824
825 shard->clipModel->SetId( clipModelId );
826
827 BecomeActive( TH_PHYSICS );
828}

831================
832idBrittleFracture::Shatter
833================
834*/
835void idBrittleFracture::Shatter( const idVec3 &point, const idVec3 &impulse, const int time ) {
836 int i;
837 idVec3 dir;
838 shard_t *shard;
839 float m;
840
841 if ( gameLocal.isServer ) {
842 idBitMsg msg;
843 byte msgBuf[MAX_EVENT_PARAM_SIZE];
844
845 msg.Init( msgBuf, sizeof( msgBuf ) );
846 msg.BeginWriting();
847 msg.WriteFloat( point[0] );
848 msg.WriteFloat( point[1] );
849 msg.WriteFloat( point[2] );
850 msg.WriteFloat( impulse[0] );
851 msg.WriteFloat( impulse[1] );
852 msg.WriteFloat( impulse[2] );
853 ServerSendEvent( EVENT_SHATTER, &msg, true, -1 );
854 }
855
856 if ( time > ( gameLocal.time - SHARD_ALIVE_TIME ) ) {
857 StartSound( "snd_shatter", SND_CHANNEL_ANY, 0, false, NULL );
858 }
859
860 if ( !IsBroken() ) {
861 Break();
862 }
863
864 if ( fxFracture.Length() ) {
865 idEntityFx::StartFx( fxFracture, &point, &GetPhysics()->GetAxis(), this, true );
866 }
867
868 dir = impulse;
869 m = dir.Normalize();
870
871 for ( i = 0; i < shards.Num(); i++ ) {
872 shard = shards[i];
873
874 if ( shard->droppedTime != -1 ) {
875 continue;
876 }
877
878 if ( ( shard->clipModel->GetOrigin() - point ).LengthSqr() > Square( maxShatterRadius ) ) {
879 continue;
880 }
881
882 DropShard( shard, point, dir, m, time );
883 }
884
885 DropFloatingIslands( point, impulse, time );
886}

888/*
889================
890idBrittleFracture::DropFloatingIslands
891================
892*/
893void idBrittleFracture::DropFloatingIslands( const idVec3 &point, const idVec3 &impulse, const int time ) {
894 int i, j, numIslands;
895 int queueStart, queueEnd;
896 shard_t *curShard, *nextShard, **queue;
897 bool touchesEdge;
898 idVec3 dir;
899
900 dir = impulse;
901 dir.Normalize();
902
903 numIslands = 0;
904 queue = (shard_t **) _alloca16( shards.Num() * sizeof(shard_t **) );
905 for ( i = 0; i < shards.Num(); i++ ) {
906 shards[i]->islandNum = 0;
907 }
908
909 for ( i = 0; i < shards.Num(); i++ ) {
910
911 if ( shards[i]->droppedTime != -1 ) {
912 continue;
913 }
914
915 if ( shards[i]->islandNum ) {
916 continue;
917 }
918
919 queueStart = 0;
920 queueEnd = 1;
921 queue[0] = shards[i];
922 shards[i]->islandNum = numIslands+1;
923 touchesEdge = false;
924
925 if ( shards[i]->atEdge ) {
926 touchesEdge = true;
927 }
928
929 for ( curShard = queue[queueStart]; queueStart < queueEnd; curShard = queue[++queueStart] ) {
930
931 for ( j = 0; j < curShard->neighbours.Num(); j++ ) {
932
933 nextShard = curShard->neighbours[j];
934
935 if ( nextShard->droppedTime != -1 ) {
936 continue;
937 }
938
939 if ( nextShard->islandNum ) {
940 continue;
941 }
942
943 queue[queueEnd++] = nextShard;
944 nextShard->islandNum = numIslands+1;
945
946 if ( nextShard->atEdge ) {
947 touchesEdge = true;
948 }
949 }
950 }
951 numIslands++;
952
953 // if the island is not connected to the world at any edges
954 if ( !touchesEdge ) {
955 for ( j = 0; j < queueEnd; j++ ) {
956 DropShard( queue[j], point, dir, 0.0f, time );
957 }
958 }
959 }

1004/*
1005================
1006idBrittleFracture::Fracture_r
1007================
1008*/
1009void idBrittleFracture::Fracture_r( idFixedWinding &w ) {
1010 int i, j, bestPlane;
1011 float a, c, s, dist, bestDist;
1012 idVec3 origin;
1013 idPlane windingPlane, splitPlanes[2];
1014 idMat3 axis, axistemp;
1015 idFixedWinding back;
1016 idTraceModel trm;
1017 idClipModel *clipModel;
1018
1019 while( 1 ) {
1020 origin = w.GetCenter();
1021 w.GetPlane( windingPlane );
1022
1023 if ( w.GetArea() < maxShardArea ) {
1024 break;
1025 }
1026
1027 // randomly create a split plane
1028 a = gameLocal.random.RandomFloat() * idMath::TWO_PI;
1029 c = cos( a );
1030 s = -sin( a );
1031 axis[2] = windingPlane.Normal();
1032 axis[2].NormalVectors( axistemp[0], axistemp[1] );
1033 axis[0] = axistemp[ 0 ] * c + axistemp[ 1 ] * s;
1034 axis[1] = axistemp[ 0 ] * s + axistemp[ 1 ] * -c;
1035
1036 // get the best split plane
1037 bestDist = 0.0f;
1038 bestPlane = 0;
1039 for ( i = 0; i < 2; i++ ) {
1040 splitPlanes[i].SetNormal( axis[i] );
1041 splitPlanes[i].FitThroughPoint( origin );
1042 for ( j = 0; j < w.GetNumPoints(); j++ ) {
1043 dist = splitPlanes[i].Distance( w[j].ToVec3() );
1044 if ( dist > bestDist ) {
1045 bestDist = dist;
1046 bestPlane = i;
1047 }
1048 }
1049 }
1050
1051 // split the winding
1052 if ( !w.Split( &back, splitPlanes[bestPlane] ) ) {
1053 break;
1054 }
1055
1056 // recursively create shards for the back winding
1057 Fracture_r( back );
1058 }
1059
1060 // translate the winding to it's center
1061 origin = w.GetCenter();
1062 for ( j = 0; j < w.GetNumPoints(); j++ ) {
1063 w[j].ToVec3() -= origin;
1064 }
1065 w.RemoveEqualPoints();
1066
1067 trm.SetupPolygon( w );
1068 trm.Shrink( CM_CLIP_EPSILON );
1069 clipModel = new idClipModel( trm );
1070
1071 physicsObj.SetClipModel( clipModel, 1.0f, shards.Num() );
1072 physicsObj.SetOrigin( GetPhysics()->GetOrigin() + origin, shards.Num() );
1073 physicsObj.SetAxis( GetPhysics()->GetAxis(), shards.Num() );
1074
1075 AddShard( clipModel, w );
1076}

1117/*
1118================
1119idBrittleFracture::FindNeighbours
1120================
1121*/
1122void idBrittleFracture::FindNeighbours( void ) {
1123 int i, j, k, l;
1124 idVec3 p1, p2, dir;
1125 idMat3 axis;
1126 idPlane plane[4];
1127
1128 for ( i = 0; i < shards.Num(); i++ ) {
1129
1130 shard_t *shard1 = shards[i];
1131 const idWinding &w1 = shard1->winding;
1132 const idVec3 &origin1 = shard1->clipModel->GetOrigin();
1133 const idMat3 &axis1 = shard1->clipModel->GetAxis();
1134
1135 for ( k = 0; k < w1.GetNumPoints(); k++ ) {
1136
1137 p1 = origin1 + w1[k].ToVec3() * axis1;
1138 p2 = origin1 + w1[(k+1)%w1.GetNumPoints()].ToVec3() * axis1;
1139 dir = p2 - p1;
1140 dir.Normalize();
1141 axis = dir.ToMat3();
1142
1143 plane[0].SetNormal( dir );
1144 plane[0].FitThroughPoint( p1 );
1145 plane[1].SetNormal( -dir );
1146 plane[1].FitThroughPoint( p2 );
1147 plane[2].SetNormal( axis[1] );
1148 plane[2].FitThroughPoint( p1 );
1149 plane[3].SetNormal( axis[2] );
1150 plane[3].FitThroughPoint( p1 );
1151
1152 for ( j = 0; j < shards.Num(); j++ ) {
1153
1154 if ( i == j ) {
1155 continue;
1156 }
1157
1158 shard_t *shard2 = shards[j];
1159
1160 for ( l = 0; l < shard1->neighbours.Num(); l++ ) {
1161 if ( shard1->neighbours[l] == shard2 ) {
1162 break;
1163 }
1164 }
1165 if ( l < shard1->neighbours.Num() ) {
1166 continue;
1167 }
1168
1169 const idWinding &w2 = shard2->winding;
1170 const idVec3 &origin2 = shard2->clipModel->GetOrigin();
1171 const idMat3 &axis2 = shard2->clipModel->GetAxis();
1172
1173 for ( l = w2.GetNumPoints()-1; l >= 0; l-- ) {
1174 p1 = origin2 + w2[l].ToVec3() * axis2;
1175 p2 = origin2 + w2[(l-1+w2.GetNumPoints())%w2.GetNumPoints()].ToVec3() * axis2;
1176 if ( plane[0].Side( p2, 0.1f ) == SIDE_FRONT && plane[1].Side( p1, 0.1f ) == SIDE_FRONT ) {
1177 if ( plane[2].Side( p1, 0.1f ) == SIDE_ON && plane[3].Side( p1, 0.1f ) == SIDE_ON ) {
1178 if ( plane[2].Side( p2, 0.1f ) == SIDE_ON && plane[3].Side( p2, 0.1f ) == SIDE_ON ) {
1179 shard1->neighbours.Append( shard2 );
1180 shard1->edgeHasNeighbour[k] = true;
1181 shard2->neighbours.Append( shard1 );
1182 shard2->edgeHasNeighbour[(l-1+w2.GetNumPoints())%w2.GetNumPoints()] = true;
1183 break;
1184 }
1185 }
1186 }
1187 }
1188 }
1189 }
1190
1191 for ( k = 0; k < w1.GetNumPoints(); k++ ) {
1192 if ( !shard1->edgeHasNeighbour[k] ) {
1193 break;
1194 }
1195 }
1196 if ( k < w1.GetNumPoints() ) {
1197 shard1->atEdge = true;
1198 } else {
1199 shard1->atEdge = false;
1200 }
1201 }
1202}

204 p->tourneyRank++;
205 }
206 playerState[ clientNum ].ingame = ingame;

286/*
287================
288idMultiplayerGame::UpdatePlayerRanks
289================
290*/
291void idMultiplayerGame::UpdatePlayerRanks() {
292 int i, j, k;
293 idPlayer *players[MAX_CLIENTS];
294 idEntity *ent;
295 idPlayer *player;
296
297 memset( players, 0, sizeof( players ) );
298 numRankedPlayers = 0;
299
300 for ( i = 0; i < gameLocal.numClients; i++ ) {
301 ent = gameLocal.entities[ i ];
302 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
303 continue;
304 }
305 player = static_cast< idPlayer * >( ent );
306 if ( !CanPlay( player ) ) {
307 continue;
308 }
309 if ( gameLocal.gameType == GAME_TOURNEY ) {
310 if ( i != currentTourneyPlayer[ 0 ] && i != currentTourneyPlayer[ 1 ] ) {
311 continue;
312 }
313 }
314 if ( gameLocal.gameType == GAME_LASTMAN && playerState[ i ].fragCount == LASTMAN_NOLIVES ) {
315 continue;
316 }
317 for ( j = 0; j < numRankedPlayers; j++ ) {
318 bool insert = false;
319 if ( gameLocal.gameType == GAME_TDM ) {
320 if ( player->team != players[ j ]->team ) {
321 if ( playerState[ i ].teamFragCount > playerState[ players[ j ]->entityNumber ].teamFragCount ) {
322 // team scores
323 insert = true;
324 } else if ( playerState[ i ].teamFragCount == playerState[ players[ j ]->entityNumber ].teamFragCount && player->team < players[ j ]->team ) {
325 // at equal scores, sort by team number
326 insert = true;
327 }
328 } else if ( playerState[ i ].fragCount > playerState[ players[ j ]->entityNumber ].fragCount ) {
329 // in the same team, sort by frag count
330 insert = true;
331 }
332 } else {
333 insert = ( playerState[ i ].fragCount > playerState[ players[ j ]->entityNumber ].fragCount );
334 }
335 if ( insert ) {
336 for ( k = numRankedPlayers; k > j; k-- ) {
337 players[ k ] = players[ k-1 ];
338 }
339 players[ j ] = player;
340 break;
341 }
342 }
343 if ( j == numRankedPlayers ) {
344 players[ numRankedPlayers ] = player;
345 }
346 numRankedPlayers++;
347 }
348
349 memcpy( rankedPlayers, players, sizeof( players ) );
350}

365================
366idMultiplayerGame::UpdateScoreboard
367================
368*/
369void idMultiplayerGame::UpdateScoreboard( idUserInterface *scoreBoard, idPlayer *player ) {
370 int i, j, iline, k;
371 idStr gameinfo;
372 idStr livesinfo;
373 idStr timeinfo;
374 idEntity *ent;
375 idPlayer *p;
376 int value;
377
378 scoreBoard->SetStateString( "scoretext", gameLocal.gameType == GAME_LASTMAN ? common->GetLanguageDict()->GetString( "#str_04242" ) : common->GetLanguageDict()->GetString( "#str_04243" ) );
379
380 iline = 0; // the display lines
381 if ( gameState != WARMUP ) {
382 for ( i = 0; i < numRankedPlayers; i++ ) {
383 // ranked player
384 iline++;
385 scoreBoard->SetStateString( va( "player%i", iline ), rankedPlayers[ i ]->GetUserInfo()->GetString( "ui_name" ) );
386 if ( gameLocal.gameType == GAME_TDM ) {
387 value = idMath::ClampInt( MP_PLAYER_MINFRAGS, MP_PLAYER_MAXFRAGS, playerState[ rankedPlayers[ i ]->entityNumber ].fragCount );
388 scoreBoard->SetStateInt( va( "player%i_tdm_score", iline ), value );
389 value = idMath::ClampInt( MP_PLAYER_MINFRAGS, MP_PLAYER_MAXFRAGS, playerState[ rankedPlayers[ i ]->entityNumber ].teamFragCount );
390 scoreBoard->SetStateString( va( "player%i_tdm_tscore", iline ), va( "/ %i", value ) );
391 scoreBoard->SetStateString( va( "player%i_score", iline ), "" );
392 } else {
393 value = idMath::ClampInt( MP_PLAYER_MINFRAGS, MP_PLAYER_MAXFRAGS, playerState[ rankedPlayers[ i ]->entityNumber ].fragCount );
394 scoreBoard->SetStateInt( va( "player%i_score", iline ), value );
395 scoreBoard->SetStateString( va( "player%i_tdm_tscore", iline ), "" );
396 scoreBoard->SetStateString( va( "player%i_tdm_score", iline ), "" );
397 }
398 value = idMath::ClampInt( 0, MP_PLAYER_MAXWINS, playerState[ rankedPlayers[ i ]->entityNumber ].wins );
399 scoreBoard->SetStateInt( va( "player%i_wins", iline ), value );
400 scoreBoard->SetStateInt( va( "player%i_ping", iline ), playerState[ rankedPlayers[ i ]->entityNumber ].ping );
401 // set the color band
402 scoreBoard->SetStateInt( va( "rank%i", iline ), 1 );
403 UpdateRankColor( scoreBoard, "rank%i_color%i", iline, rankedPlayers[ i ]->colorBar );
404 if ( rankedPlayers[ i ] == player ) {
405 // highlight who we are
406 scoreBoard->SetStateInt( "rank_self", iline );
407 }
408 }
409 }
410
411 // if warmup, this draws everyone, otherwise it goes over spectators only
412 // when doing warmup we loop twice to draw ready/not ready first *then* spectators
413 // NOTE: in tourney, shows spectators according to their playing rank order?
414 for ( k = 0; k < ( gameState == WARMUP ? 2 : 1 ); k++ ) {
415 for ( i = 0; i < MAX_CLIENTS; i++ ) {
416 ent = gameLocal.entities[ i ];
417 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
418 continue;
419 }
420 if ( gameState != WARMUP ) {
421 // check he's not covered by ranks already
422 for ( j = 0; j < numRankedPlayers; j++ ) {
423 if ( ent == rankedPlayers[ j ] ) {
424 break;
425 }
426 }
427 if ( j != numRankedPlayers ) {
428 continue;
429 }
430 }
431 p = static_cast< idPlayer * >( ent );
432 if ( gameState == WARMUP ) {
433 if ( k == 0 && p->spectating ) {
434 continue;
435 }
436 if ( k == 1 && !p->spectating ) {
437 continue;
438 }
439 }
440
441 iline++;
442 if ( !playerState[ i ].ingame ) {
443 scoreBoard->SetStateString( va( "player%i", iline ), common->GetLanguageDict()->GetString( "#str_04244" ) );
444 scoreBoard->SetStateString( va( "player%i_score", iline ), common->GetLanguageDict()->GetString( "#str_04245" ) );
445 // no color band
446 scoreBoard->SetStateInt( va( "rank%i", iline ), 0 );
447 } else {
448 scoreBoard->SetStateString( va( "player%i", iline ), gameLocal.userInfo[ i ].GetString( "ui_name" ) );
449 if ( gameState == WARMUP ) {
450 if ( p->spectating ) {
451 scoreBoard->SetStateString( va( "player%i_score", iline ), common->GetLanguageDict()->GetString( "#str_04246" ) );
452 // no color band
453 scoreBoard->SetStateInt( va( "rank%i", iline ), 0 );
454 } else {
455 scoreBoard->SetStateString( va( "player%i_score", iline ), p->IsReady() ? common->GetLanguageDict()->GetString( "#str_04247" ) : common->GetLanguageDict()->GetString( "#str_04248" ) );
456 // set the color band
457 scoreBoard->SetStateInt( va( "rank%i", iline ), 1 );
458 UpdateRankColor( scoreBoard, "rank%i_color%i", iline, p->colorBar );
459 }
460 } else {
461 if ( gameLocal.gameType == GAME_LASTMAN && playerState[ i ].fragCount == LASTMAN_NOLIVES ) {
462 scoreBoard->SetStateString( va( "player%i_score", iline ), common->GetLanguageDict()->GetString( "#str_06736" ) );
463 // set the color band
464 scoreBoard->SetStateInt( va( "rank%i", iline ), 1 );
465 UpdateRankColor( scoreBoard, "rank%i_color%i", iline, p->colorBar );
466 } else {
467 scoreBoard->SetStateString( va( "player%i_score", iline ), common->GetLanguageDict()->GetString( "#str_04246" ) );
468 // no color band
469 scoreBoard->SetStateInt( va( "rank%i", iline ), 0 );
470 }
471 }
472 }
473 scoreBoard->SetStateString( va( "player%i_tdm_tscore", iline ), "" );
474 scoreBoard->SetStateString( va( "player%i_tdm_score", iline ), "" );
475 scoreBoard->SetStateString( va( "player%i_wins", iline ), "" );
476 scoreBoard->SetStateInt( va( "player%i_ping", iline ), playerState[ i ].ping );
477 if ( i == player->entityNumber ) {
478 // highlight who we are
479 scoreBoard->SetStateInt( "rank_self", iline );
480 }
481 }
482 }
483
484 // clear remaining lines (empty slots)
485 iline++;
486 while ( iline < 5 ) {
487 scoreBoard->SetStateString( va( "player%i", iline ), "" );
488 scoreBoard->SetStateString( va( "player%i_score", iline ), "" );
489 scoreBoard->SetStateString( va( "player%i_tdm_tscore", iline ), "" );
490 scoreBoard->SetStateString( va( "player%i_tdm_score", iline ), "" );
491 scoreBoard->SetStateString( va( "player%i_wins", iline ), "" );
492 scoreBoard->SetStateString( va( "player%i_ping", iline ), "" );
493 scoreBoard->SetStateInt( va( "rank%i", iline ), 0 );
494 iline++;
495 }
496
497 gameinfo = va( "%s: %s", common->GetLanguageDict()->GetString( "#str_02376" ), gameLocal.serverInfo.GetString( "si_gameType" ) );
498 if ( gameLocal.gameType == GAME_LASTMAN ) {
499 if ( gameState == GAMEON || gameState == SUDDENDEATH ) {
500 livesinfo = va( "%s: %i", common->GetLanguageDict()->GetString( "#str_04264" ), startFragLimit );
501 } else {
502 livesinfo = va( "%s: %i", common->GetLanguageDict()->GetString( "#str_04264" ), gameLocal.serverInfo.GetInt( "si_fragLimit" ) );
503 }
504
505 } else {
506 livesinfo = va( "%s: %i", common->GetLanguageDict()->GetString( "#str_01982" ), gameLocal.serverInfo.GetInt( "si_fragLimit" ) );
507 }
508 if ( gameLocal.serverInfo.GetInt( "si_timeLimit" ) > 0 ) {
509 timeinfo = va( "%s: %i", common->GetLanguageDict()->GetString( "#str_01983" ), gameLocal.serverInfo.GetInt( "si_timeLimit" ) );
510 } else {
511 timeinfo = va("%s", common->GetLanguageDict()->GetString( "#str_07209" ));
512 }
513 scoreBoard->SetStateString( "gameinfo", gameinfo );
514 scoreBoard->SetStateString( "livesinfo", livesinfo );
515 scoreBoard->SetStateString( "timeinfo", timeinfo );
516
517 scoreBoard->Redraw( gameLocal.time );
518}

520/*
521================
522idMultiplayerGame::GameTime
523================
524*/
525const char *idMultiplayerGame::GameTime() {
526 static char buff[16];
527 int m, s, t, ms;
528
529 if ( gameState == COUNTDOWN ) {
530 ms = warmupEndTime - gameLocal.realClientTime;
531 s = ms / 1000 + 1;
532 if ( ms <= 0 ) {
533 strcpy( buff, "WMP --" );
534 } else {
535 sprintf( buff, "WMP %i", s );
536 }
537 } else {
538 int timeLimit = gameLocal.serverInfo.GetInt( "si_timeLimit" );
539 if ( timeLimit ) {
540 ms = ( timeLimit * 60000 ) - ( gameLocal.time - matchStartedTime );
541 } else {
542 ms = gameLocal.time - matchStartedTime;
543 }
544 if ( ms < 0 ) {
545 ms = 0;
546 }
547
548 s = ms / 1000;
549 m = s / 60;
550 s -= m * 60;
551 t = s / 10;
552 s -= t * 10;
553
554 sprintf( buff, "%i:%i%i", m, t, s );
555 }
556 return &buff[0];
557}

561idMultiplayerGame::NumActualClients
562================
563*/
564int idMultiplayerGame::NumActualClients( bool countSpectators, int *teamcounts ) {
565 idPlayer *p;
566 int c = 0;
567
568 if ( teamcounts ) {
569 teamcounts[ 0 ] = teamcounts[ 1 ] = 0;
570 }
571 for( int i = 0 ; i < gameLocal.numClients ; i++ ) {
572 idEntity *ent = gameLocal.entities[ i ];
573 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
574 continue;
575 }
576 p = static_cast< idPlayer * >( ent );
577 if ( countSpectators || CanPlay( p ) ) {
578 c++;
579 }
580 if ( teamcounts && CanPlay( p ) ) {
581 teamcounts[ p->team ]++;
582 }
583 }
584 return c;
585}

603================
604idMultiplayerGame::AllPlayersReady
605================
606*/
607bool idMultiplayerGame::AllPlayersReady() {
608 int i;
609 idEntity *ent;
610 idPlayer *p;
611 int team[ 2 ];
612
613 if ( NumActualClients( false, &team[ 0 ] ) <= 1 ) {
614 return false;
615 }
616
617 if ( gameLocal.gameType == GAME_TDM ) {
618 if ( !team[ 0 ] || !team[ 1 ] ) {
619 return false;
620 }
621 }
622
623 if ( !gameLocal.serverInfo.GetBool( "si_warmup" ) ) {
624 return true;
625 }
626
627 for( i = 0; i < gameLocal.numClients; i++ ) {
628 if ( gameLocal.gameType == GAME_TOURNEY && i != currentTourneyPlayer[ 0 ] && i != currentTourneyPlayer[ 1 ] ) {
629 continue;
630 }
631 ent = gameLocal.entities[ i ];
632 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
633 continue;
634 }
635 p = static_cast< idPlayer * >( ent );
636 if ( CanPlay( p ) && !p->IsReady() ) {
637 return false;
638 }
639 team[ p->team ]++;
640 }
641
642 return true;
643}

645/*
646================
647idMultiplayerGame::FragLimitHit
648return the winning player (team player)
649if there is no FragLeader(), the game is tied and we return NULL
650================
651*/
652idPlayer *idMultiplayerGame::FragLimitHit() {
653 int i;
654 int fragLimit = gameLocal.serverInfo.GetInt( "si_fragLimit" );
655 idPlayer *leader;
656
657 leader = FragLeader();
658 if ( !leader ) {
659 return NULL;
660 }
661
662 if ( fragLimit <= 0 ) {
663 fragLimit = MP_PLAYER_MAXFRAGS;
664 }
665
666 if ( gameLocal.gameType == GAME_LASTMAN ) {
667 // we have a leader, check if any other players have frags left
668 assert( !static_cast< idPlayer * >( leader )->lastManOver );
669 for( i = 0 ; i < gameLocal.numClients ; i++ ) {
670 idEntity *ent = gameLocal.entities[ i ];
671 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
672 continue;
673 }
674 if ( !CanPlay( static_cast< idPlayer * >( ent ) ) ) {
675 continue;
676 }
677 if ( ent == leader ) {
678 continue;
679 }
680 if ( playerState[ ent->entityNumber ].fragCount > 0 ) {
681 return NULL;
682 }
683 }
684 // there is a leader, his score may even be negative, but no one else has frags left or is !lastManOver
685 return leader;
686 } else if ( gameLocal.gameType == GAME_TDM ) {

714/*
715================
716idMultiplayerGame::FragLeader
717return the current winner ( or a player from the winning team )
718NULL if even
719================
720*/
721idPlayer *idMultiplayerGame::FragLeader( void ) {
722 int i;
723 int frags[ MAX_CLIENTS ];
724 idPlayer *leader = NULL;
725 idEntity *ent;
726 idPlayer *p;
727 int high = -9999;
728 int count = 0;
729 bool teamLead[ 2 ] = { false, false };
730
731 for ( i = 0 ; i < gameLocal.numClients ; i++ ) {
732 ent = gameLocal.entities[ i ];
733 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
734 continue;
735 }
736 if ( !CanPlay( static_cast< idPlayer * >( ent ) ) ) {
737 continue;
738 }
739 if ( gameLocal.gameType == GAME_TOURNEY && ent->entityNumber != currentTourneyPlayer[ 0 ] && ent->entityNumber != currentTourneyPlayer[ 1 ] ) {
740 continue;
741 }
742 if ( static_cast< idPlayer * >( ent )->lastManOver ) {
743 continue;
744 }
745
746 int fragc = ( gameLocal.gameType == GAME_TDM ) ? playerState[i].teamFragCount : playerState[i].fragCount;
747 if ( fragc > high ) {
748 high = fragc;
749 }
750
751 frags[ i ] = fragc;
752 }
753
754 for ( i = 0; i < gameLocal.numClients; i++ ) {
755 ent = gameLocal.entities[ i ];
756 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
757 continue;
758 }
759 p = static_cast< idPlayer * >( ent );
760 p->SetLeader( false );
761
762 if ( !CanPlay( p ) ) {
763 continue;
764 }
765 if ( gameLocal.gameType == GAME_TOURNEY && ent->entityNumber != currentTourneyPlayer[ 0 ] && ent->entityNumber != currentTourneyPlayer[ 1 ] ) {
766 continue;
767 }
768 if ( p->lastManOver ) {
769 continue;
770 }
771 if ( p->spectating ) {
772 continue;
773 }
774
775 if ( frags[ i ] >= high ) {
776 leader = p;
777 count++;
778 p->SetLeader( true );
779 if ( gameLocal.gameType == GAME_TDM ) {
780 teamLead[ p->team ] = true;
781 }
782 }
783 }
784
785 if ( gameLocal.gameType != GAME_TDM ) {
786 // more than one player at the highest frags
787 if ( count > 1 ) {
788 return NULL;
789 } else {
790 return leader;
791 }
792 } else {
793 if ( teamLead[ 0 ] && teamLead[ 1 ] ) {
794 // even game in team play
795 return NULL;
796 }
797 return leader;
798 }
799}

973 idBitMsg outMsg;
974 byte msgBuf[MAX_GAME_MESSAGE_SIZE];
975 int i;
976
977 assert( news != gameState );
978 assert( !gameLocal.isClient );
979 gameLocal.DPrintf( "%s -> %s\n", GameStateStrings[ gameState ], GameStateStrings[ news ] );
980 switch( news ) {
981 case GAMEON: {
982 gameLocal.LocalMapRestart();
983 outMsg.Init( msgBuf, sizeof( msgBuf ) );
984 outMsg.WriteByte( GAME_RELIABLE_MESSAGE_RESTART );

991 for( i = 0; i < gameLocal.numClients; i++ ) {
992 idEntity *ent = gameLocal.entities[ i ];
993 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
994 continue;
995 }
996 idPlayer *p = static_cast<idPlayer *>( ent );
997 p->SetLeader( false ); // don't carry the flag from previous games
998 if ( gameLocal.gameType == GAME_TOURNEY && currentTourneyPlayer[ 0 ] != i && currentTourneyPlayer[ 1 ] != i ) {
999 p->ServerSpectate( true );
1000 p->tourneyRank++;
1001 } else {

1007 static_cast<idPlayer *>(ent)->ServerSpectate( false );
1008 if ( gameLocal.gameType == GAME_TOURNEY ) {
1009 p->tourneyRank = 0;
1010 }
1011 }
1012 }

1014 p->lastManPresent = true;
1015 } else {
1016 p->lastManPresent = false;
1017 }
1018 }

1020 switchThrottle[ 1 ] = 0; // passby the throttle

1022 break;
1023 }
1024 case GAMEREVIEW: {

1027 for( i = 0; i < gameLocal.numClients; i++ ) {
1028 idEntity *ent = gameLocal.entities[ i ];
1029 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
1030 continue;

1033 static_cast<idPlayer *>(ent)->ServerSpectate( true );
1034 }
1035 UpdateWinsLosses( player );
1036 break;
1037 }
1038 case SUDDENDEATH: {
1039 PrintMessageEvent( -1, MSG_SUDDENDEATH );
1040 PlayGlobalSound( -1, SND_SUDDENDEATH );
1041 break;
1042 }
1043 case COUNTDOWN: {

1053
1054 break;
1055 }
1056 default:
1057 break;
1058 }
1059
1060 gameState = news;

1291 ClientUpdateVote( VOTE_FAILED, yesVotes, noVotes );

1293 return;

1306/*
1307================
1308idMultiplayerGame::Run
1309================
1310*/
1311void idMultiplayerGame::Run() {
1312 int i, timeLeft;
1313 idPlayer *player;
1314 int gameReviewPause;
1315
1316 assert( gameLocal.isMultiplayer );
1317 assert( !gameLocal.isClient );
1318
1319 pureReady = true;
1320
1321 if ( gameState == INACTIVE ) {
1322 lastGameType = gameLocal.gameType;
1323 NewState( WARMUP );
1324 }
1325
1326 CheckVote();
1327
1328 CheckRespawns();
1329
1330 if ( nextState != INACTIVE && gameLocal.time > nextStateSwitch ) {
1331 NewState( nextState );
1332 nextState = INACTIVE;
1333 }
1334
1335 // don't update the ping every frame to save bandwidth
1336 if ( gameLocal.time > pingUpdateTime ) {
1337 for ( i = 0; i < gameLocal.numClients; i++ ) {
1338 playerState[i].ping = networkSystem->ServerGetClientPing( i );
1339 }
1340 pingUpdateTime = gameLocal.time + 1000;
1341 }
1342
1343 warmupText = "";
1344
1345 switch( gameState ) {
1346 case GAMEREVIEW: {
1347 if ( nextState == INACTIVE ) {
1348 gameReviewPause = cvarSystem->GetCVarInteger( "g_gameReviewPause" );
1349 nextState = NEXTGAME;
1350 nextStateSwitch = gameLocal.time + 1000 * gameReviewPause;
1351 }
1352 break;
1353 }
1354 case NEXTGAME: {
1355 if ( nextState == INACTIVE ) {
1356 // game rotation, new map, gametype etc.
1357 if ( gameLocal.NextMap() ) {
1358 cmdSystem->BufferCommandText( CMD_EXEC_APPEND, "serverMapRestart\n" );
1359 return;
1360 }
1361 NewState( WARMUP );
1362 if ( gameLocal.gameType == GAME_TOURNEY ) {
1363 CycleTourneyPlayers();
1364 }
1365 // put everyone back in from endgame spectate
1366 for ( i = 0; i < gameLocal.numClients; i++ ) {
1367 idEntity *ent = gameLocal.entities[ i ];
1368 if ( ent && ent->IsType( idPlayer::Type ) ) {
1369 if ( !static_cast< idPlayer * >( ent )->wantSpectate ) {
1370 CheckRespawns( static_cast<idPlayer *>( ent ) );
1371 }
1372 }
1373 }
1374 }
1375 break;
1376 }
1377 case WARMUP: {
1378 if ( AllPlayersReady() ) {
1379 NewState( COUNTDOWN );
1380 nextState = GAMEON;
1381 nextStateSwitch = gameLocal.time + 1000 * cvarSystem->GetCVarInteger( "g_countDown" );
1382 }
1383 warmupText = "Warming up.. waiting for players to get ready";
1384 one = two = three = false;
1385 break;
1386 }
1387 case COUNTDOWN: {
1388 timeLeft = ( nextStateSwitch - gameLocal.time ) / 1000 + 1;
1389 if ( timeLeft == 3 && !three ) {
1390 PlayGlobalSound( -1, SND_THREE );
1391 three = true;
1392 } else if ( timeLeft == 2 && !two ) {
1393 PlayGlobalSound( -1, SND_TWO );
1394 two = true;
1395 } else if ( timeLeft == 1 && !one ) {
1396 PlayGlobalSound( -1, SND_ONE );
1397 one = true;
1398 }
1399 warmupText = va( "Match starts in %i", timeLeft );
1400 break;
1401 }
1402 case GAMEON: {
1403 player = FragLimitHit();
1404 if ( player ) {
1405 // delay between detecting frag limit and ending game. let the death anims play
1406 if ( !fragLimitTimeout ) {
1407 common->DPrintf( "enter FragLimit timeout, player %d is leader\n", player->entityNumber );
1408 fragLimitTimeout = gameLocal.time + FRAGLIMIT_DELAY;
1409 }
1410 if ( gameLocal.time > fragLimitTimeout ) {
1411 NewState( GAMEREVIEW, player );
1412 PrintMessageEvent( -1, MSG_FRAGLIMIT, player->entityNumber );
1413 }
1414 } else {
1415 if ( fragLimitTimeout ) {
1416 // frag limit was hit and cancelled. means the two teams got even during FRAGLIMIT_DELAY
1417 // enter sudden death, the next frag leader will win
1418 SuddenRespawn();
1419 PrintMessageEvent( -1, MSG_HOLYSHIT );
1420 fragLimitTimeout = 0;
1421 NewState( SUDDENDEATH );
1422 } else if ( TimeLimitHit() ) {
1423 player = FragLeader();
1424 if ( !player ) {
1425 NewState( SUDDENDEATH );
1426 } else {
1427 NewState( GAMEREVIEW, player );
1428 PrintMessageEvent( -1, MSG_TIMELIMIT );
1429 }
1430 }
1431 }
1432 break;
1433 }
1434 case SUDDENDEATH: {
1435 player = FragLeader();
1436 if ( player ) {
1437 if ( !fragLimitTimeout ) {
1438 common->DPrintf( "enter sudden death FragLeader timeout, player %d is leader\n", player->entityNumber );
1439 fragLimitTimeout = gameLocal.time + FRAGLIMIT_DELAY;
1440 }
1441 if ( gameLocal.time > fragLimitTimeout ) {
1442 NewState( GAMEREVIEW, player );
1443 PrintMessageEvent( -1, MSG_FRAGLIMIT, player->entityNumber );
1444 }
1445 } else if ( fragLimitTimeout ) {
1446 SuddenRespawn();
1447 PrintMessageEvent( -1, MSG_HOLYSHIT );
1448 fragLimitTimeout = 0;
1449 }
1450 break;
1451 }
1452 }
1453}

2301void idMultiplayerGame::CheckRespawns( idPlayer *spectator ) {

2303 idEntity *ent = gameLocal.entities[ i ];
2304 if ( !ent || !ent->IsType( idPlayer::Type ) ) {
2305 continue;
2306 }
2307 idPlayer *p = static_cast<idPlayer *>(ent);
2308 // once we hit sudden death, nobody respawns till game has ended
2309 if ( WantRespawn( p ) || p == spectator ) {
2310 if ( gameState == SUDDENDEATH && gameLocal.gameType != GAME_LASTMAN ) {
2311 // respawn rules while sudden death are different
2312 // sudden death may trigger while a player is dead, so there are still cases where we need to respawn
2313 // don't do any respawns while we are in end game delay though
2314 if ( !fragLimitTimeout ) {
2315 if ( gameLocal.gameType == GAME_TDM || p->IsLeader() ) {
2316#ifdef _DEBUG
2317 if ( gameLocal.gameType == GAME_TOURNEY ) {
2318 assert( p->entityNumber == currentTourneyPlayer[ 0 ] || p->entityNumber == currentTourneyPlayer[ 1 ] );
2319 }
2320#endif
2321 p->ServerSpectate( false );
2322 } else if ( !p->IsLeader() ) {
2323 // sudden death is rolling, this player is not a leader, have him spectate
2324 p->ServerSpectate( true );
2325 CheckAbortGame();
2326 }
2327 }
2328 } else {
2329 if ( gameLocal.gameType == GAME_DM ||
2330 gameLocal.gameType == GAME_TDM ) {
2331 if ( gameState == WARMUP || gameState == COUNTDOWN || gameState == GAMEON ) {
2332 p->ServerSpectate( false );
2333 }
2334 } else if ( gameLocal.gameType == GAME_TOURNEY ) {
2335 if ( i == currentTourneyPlayer[ 0 ] || i == currentTourneyPlayer[ 1 ] ) {
2336 if ( gameState == WARMUP || gameState == COUNTDOWN || gameState == GAMEON ) {
2337 p->ServerSpectate( false );
2338 }
2339 } else if ( gameState == WARMUP ) {
2340 // make sure empty tourney slots get filled first
2341 FillTourneySlots( );
2342 if ( i == currentTourneyPlayer[ 0 ] || i == currentTourneyPlayer[ 1 ] ) {
2343 p->ServerSpectate( false );
2344 }
2345 }
2346 } else if ( gameLocal.gameType == GAME_LASTMAN ) {
2347 if ( gameState == WARMUP || gameState == COUNTDOWN ) {
2348 p->ServerSpectate( false );
2349 } else if ( gameState == GAMEON || gameState == SUDDENDEATH ) {
2350 if ( gameState == GAMEON && playerState[ i ].fragCount > 0 && p->lastManPresent ) {
2351 assert( !p->lastManOver );
2352 p->ServerSpectate( false );
2353 } else if ( p->lastManPlayAgain && p->lastManPresent ) {
2354 assert( gameState == SUDDENDEATH );
2355 p->ServerSpectate( false );
2356 } else {
2357 // if a fragLimitTimeout was engaged, do NOT mark lastManOver as that could mean
2358 // everyone ends up spectator and game is stalled with no end
2359 // if the frag limit delay is engaged and cancels out before expiring, LMN players are
2360 // respawned to play the tie again ( through SuddenRespawn and lastManPlayAgain )
2361 if ( !fragLimitTimeout && !p->lastManOver ) {
2362 common->DPrintf( "client %d has lost all last man lives\n", i );

2369

2372 //and the server got stuck

2374 int j;
2375 for ( j = 0; j < gameLocal.numClients; j++ ) {
2376 if ( !gameLocal.entities[ j ] ) {
2377 continue;
2378 }
2379 if ( !CanPlay( static_cast< idPlayer * >( gameLocal.entities[ j ] ) ) ) {
2380 continue;
2381 }
2382 if ( !static_cast< idPlayer * >( gameLocal.entities[ j ] )->lastManOver ) {
2383 break;
2384 }
2385 }
2386 if( j == gameLocal.numClients) {
2387 //Everyone is dead so don't allow this player to spectate
2388 //so the match will end
2389 p->ServerSpectate( false );
2390 }
2391 }
2392 }
2393 }
2394 }
2395 }
2396 }
2397 } else if ( p->wantSpectate && !p->spectating ) {
2398 playerState[ i ].fragCount = 0; // whenever you willingly go spectate during game, your score resets
2399 p->ServerSpectate( true );

194 idDict args;

211

213
214 this->owner = owner;
215
216 memset( &renderLight, 0, sizeof( renderLight ) );
217 shaderName = spawnArgs.GetString( "mtr_light_shader" );
218 if ( *(const char *)shaderName ) {
219 renderLight.shader = declManager->FindMaterial( shaderName, false );
220 renderLight.pointLight = true;
221 renderLight.lightRadius[0] =
222 renderLight.lightRadius[1] =
223 renderLight.lightRadius[2] = spawnArgs.GetFloat( "light_radius" );
224 spawnArgs.GetVector( "light_color", "1 1 1", light_color );
225 renderLight.shaderParms[0] = light_color[0];
226 renderLight.shaderParms[1] = light_color[1];
227 renderLight.shaderParms[2] = light_color[2];
228 renderLight.shaderParms[3] = 1.0f;
229 }
230
231 spawnArgs.GetVector( "light_offset", "0 0 0", lightOffset );
232
233 lightStartTime = 0;
234 lightEndTime = 0;
235 smokeFlyTime = 0;
236
237 damagePower = 1.0f;
238
239 UpdateVisuals();
240
241 state = CREATED;
242
243 if ( spawnArgs.GetBool( "net_fullphysics" ) ) {
244 netSyncPhysics = true;
245 }
246}

270/*
271=================
272idProjectile::Launch
273=================
274*/
275void idProjectile::Launch( const idVec3 &start, const idVec3 &dir, const idVec3 &pushVelocity, const float timeSinceFire, const float launchPower, const float dmgPower ) {
276 float fuse;
277 float startthrust;
278 float endthrust;
279 idVec3 velocity;
280 idAngles angular_velocity;
281 float linear_friction;
282 float angular_friction;
283 float contact_friction;
284 float bounce;
285 float mass;
286 float speed;
287 float gravity;
288 idVec3 gravVec;
289 idVec3 tmp;
290 idMat3 axis;
291 int thrust_start;
292 int contents;
293 int clipMask;
294
295 // allow characters to throw projectiles during cinematics, but not the player
296 if ( owner.GetEntity() && !owner.GetEntity()->IsType( idPlayer::Type ) ) {
297 cinematic = owner.GetEntity()->cinematic;
298 } else {
299 cinematic = false;
300 }
301
302 thrust = spawnArgs.GetFloat( "thrust" );
303 startthrust = spawnArgs.GetFloat( "thrust_start" );
304 endthrust = spawnArgs.GetFloat( "thrust_end" );
305
306 spawnArgs.GetVector( "velocity", "0 0 0", velocity );
307
308 speed = velocity.Length() * launchPower;
309
310 damagePower = dmgPower;
311
312 spawnArgs.GetAngles( "angular_velocity", "0 0 0", angular_velocity );
313
314 linear_friction = spawnArgs.GetFloat( "linear_friction" );
315 angular_friction = spawnArgs.GetFloat( "angular_friction" );
316 contact_friction = spawnArgs.GetFloat( "contact_friction" );
317 bounce = spawnArgs.GetFloat( "bounce" );
318 mass = spawnArgs.GetFloat( "mass" );
319 gravity = spawnArgs.GetFloat( "gravity" );
320 fuse = spawnArgs.GetFloat( "fuse" );
321
322 projectileFlags.detonate_on_world = spawnArgs.GetBool( "detonate_on_world" );
323 projectileFlags.detonate_on_actor = spawnArgs.GetBool( "detonate_on_actor" );
324 projectileFlags.randomShaderSpin = spawnArgs.GetBool( "random_shader_spin" );
325
326 if ( mass <= 0 ) {
327 gameLocal.Error( "Invalid mass on '%s'\n", GetEntityDefName() );
328 }
329
330 thrust *= mass;
331 thrust_start = SEC2MS( startthrust ) + gameLocal.time;
332 thrust_end = SEC2MS( endthrust ) + gameLocal.time;
333
334 lightStartTime = 0;
335 lightEndTime = 0;
336
337 if ( health ) {
338 fl.takedamage = true;
339 }
340
341 gravVec = gameLocal.GetGravity();
342 gravVec.NormalizeFast();
343
344 Unbind();
345
346 // align z-axis of model with the direction
347 axis = dir.ToMat3();
348 tmp = axis[2];
349 axis[2] = axis[0];
350 axis[0] = -tmp;
351
352 contents = 0;
353 clipMask = MASK_SHOT_RENDERMODEL;
354 if ( spawnArgs.GetBool( "detonate_on_trigger" ) ) {
355 contents |= CONTENTS_TRIGGER;
356 }
357 if ( !spawnArgs.GetBool( "no_contents" ) ) {
358 contents |= CONTENTS_PROJECTILE;
359 clipMask |= CONTENTS_PROJECTILE;
360 }
361
362 // don't do tracers on client, we don't know origin and direction
363 if ( spawnArgs.GetBool( "tracers" ) && gameLocal.random.RandomFloat() > 0.5f ) {
364 SetModel( spawnArgs.GetString( "model_tracer" ) );
365 projectileFlags.isTracer = true;
366 }
367
368 physicsObj.SetMass( mass );
369 physicsObj.SetFriction( linear_friction, angular_friction, contact_friction );
370 if ( contact_friction == 0.0f ) {
371 physicsObj.NoContact();
372 }
373 physicsObj.SetBouncyness( bounce );
374 physicsObj.SetGravity( gravVec * gravity );
375 physicsObj.SetContents( contents );
376 physicsObj.SetClipMask( clipMask );
377 physicsObj.SetLinearVelocity( axis[ 2 ] * speed + pushVelocity );
378 physicsObj.SetAngularVelocity( angular_velocity.ToAngularVelocity() * axis );
379 physicsObj.SetOrigin( start );
380 physicsObj.SetAxis( axis );
381
382 thruster.SetPosition( &physicsObj, 0, idVec3( GetPhysics()->GetBounds()[ 0 ].x, 0, 0 ) );
383
384 if ( !gameLocal.isClient ) {
385 if ( fuse <= 0 ) {
386 // run physics for 1 second
387 RunPhysics();
388 PostEventMS( &EV_Remove, spawnArgs.GetInt( "remove_time", "1500" ) );
389 } else if ( spawnArgs.GetBool( "detonate_on_fuse" ) ) {
390 fuse -= timeSinceFire;
391 if ( fuse < 0.0f ) {
392 fuse = 0.0f;
393 }
394 PostEventSec( &EV_Explode, fuse );
395 } else {
396 fuse -= timeSinceFire;
397 if ( fuse < 0.0f ) {
398 fuse = 0.0f;
399 }
400 PostEventSec( &EV_Fizzle, fuse );
401 }
402 }
403
404 if ( projectileFlags.isTracer ) {
405 StartSound( "snd_tracer", SND_CHANNEL_BODY, 0, false, NULL );
406 } else {
407 StartSound( "snd_fly", SND_CHANNEL_BODY, 0, false, NULL );
408 }
409
410 smokeFlyTime = 0;
411 const char *smokeName = spawnArgs.GetString( "smoke_fly" );
412 if ( *smokeName != '\0' ) {
413 smokeFly = static_cast<const idDeclParticle *>( declManager->FindType( DECL_PARTICLE, smokeName ) );
414 smokeFlyTime = gameLocal.time;
415 }
416
417 // used for the plasma bolts but may have other uses as well
418 if ( projectileFlags.randomShaderSpin ) {
419 float f = gameLocal.random.RandomFloat();
420 f *= 0.5f;
421 renderEntity.shaderParms[SHADERPARM_DIVERSITY] = f;
422 }
423
424 UpdateVisuals();
425
426 state = LAUNCHED;
427}

429/*
430================
431idProjectile::Think
432================
433*/
434void idProjectile::Think( void ) {
435
436 if ( thinkFlags & TH_THINK ) {
437 if ( thrust && ( gameLocal.time < thrust_end ) ) {
438 // evaluate force
439 thruster.SetForce( GetPhysics()->GetAxis()[ 0 ] * thrust );
440 thruster.Evaluate( gameLocal.time );
441 }
442 }
443
444 // run physics
445 RunPhysics();
446
447 Present();
448
449 // add the particles
450 if ( smokeFly != NULL && smokeFlyTime && !IsHidden() ) {
451 idVec3 dir = -GetPhysics()->GetLinearVelocity();
452 dir.Normalize();
453 if ( !gameLocal.smokeParticles->EmitSmoke( smokeFly, smokeFlyTime, gameLocal.random.RandomFloat(), GetPhysics()->GetOrigin(), dir.ToMat3() ) ) {
454 smokeFlyTime = gameLocal.time;
455 }
456 }
457
458 // add the light
459 if ( renderLight.lightRadius.x > 0.0f && g_projectileLights.GetBool() ) {
460 renderLight.origin = GetPhysics()->GetOrigin() + GetPhysics()->GetAxis() * lightOffset;
461 renderLight.axis = GetPhysics()->GetAxis();
462 if ( ( lightDefHandle != -1 ) ) {
463 if ( lightEndTime > 0 && gameLocal.time <= lightEndTime + gameLocal.GetMSec() ) {
464 idVec3 color( 0, 0, 0 );
465 if ( gameLocal.time < lightEndTime ) {
466 float frac = ( float )( gameLocal.time - lightStartTime ) / ( float )( lightEndTime - lightStartTime );
467 color.Lerp( lightColor, color, frac );
468 }
469 renderLight.shaderParms[SHADERPARM_RED] = color.x;
470 renderLight.shaderParms[SHADERPARM_GREEN] = color.y;
471 renderLight.shaderParms[SHADERPARM_BLUE] = color.z;
472 }
473 gameRenderWorld->UpdateLightDef( lightDefHandle, &renderLight );
474 } else {
475 lightDefHandle = gameRenderWorld->AddLightDef( &renderLight );
476 }
477 }
478}

480/*
481=================
482idProjectile::Collide
483=================
484*/
485bool idProjectile::Collide( const trace_t &collision, const idVec3 &velocity ) {
486 idEntity *ent;
487 idEntity *ignore;
488 const char *damageDefName;
489 idVec3 dir;
490 float push;
491 float damageScale;
492
493 if ( state == EXPLODED || state == FIZZLED ) {
494 return true;
495 }
496
497 // predict the explosion
498 if ( gameLocal.isClient ) {
499 if ( ClientPredictionCollide( this, spawnArgs, collision, velocity, !spawnArgs.GetBool( "net_instanthit" ) ) ) {
500 Explode( collision, NULL );
501 return true;
502 }
503 return false;
504 }
505
506 // remove projectile when a 'noimpact' surface is hit
507 if ( ( collision.c.material != NULL ) && ( collision.c.material->GetSurfaceFlags() & SURF_NOIMPACT ) ) {
508 PostEventMS( &EV_Remove, 0 );
509 common->DPrintf( "Projectile collision no impact\n" );
510 return true;
511 }
512
513 // get the entity the projectile collided with
514 ent = gameLocal.entities[ collision.c.entityNum ];
515 if ( ent == owner.GetEntity() ) {
516 assert( 0 );
517 return true;
518 }
519
520 // just get rid of the projectile when it hits a player in noclip
521 if ( ent->IsType( idPlayer::Type ) && static_cast<idPlayer *>( ent )->noclip ) {
522 PostEventMS( &EV_Remove, 0 );
523 return true;
524 }
525
526 // direction of projectile
527 dir = velocity;
528 dir.Normalize();
529
530 // projectiles can apply an additional impulse next to the rigid body physics impulse
531 if ( spawnArgs.GetFloat( "push", "0", push ) && push > 0.0f ) {
532 ent->ApplyImpulse( this, collision.c.id, collision.c.point, push * dir );
533 }
534
535 // MP: projectiles open doors
536 if ( gameLocal.isMultiplayer && ent->IsType( idDoor::Type ) && !static_cast< idDoor * >(ent)->IsOpen() && !ent->spawnArgs.GetBool( "no_touch" ) ) {
537 ent->ProcessEvent( &EV_Activate , this );
538 }
539
540 if ( ent->IsType( idActor::Type ) || ( ent->IsType( idAFAttachment::Type ) && static_cast<const idAFAttachment*>(ent)->GetBody()->IsType( idActor::Type ) ) ) {
541 if ( !projectileFlags.detonate_on_actor ) {
542 return false;
543 }
544 } else {
545 if ( !projectileFlags.detonate_on_world ) {
546 if ( !StartSound( "snd_ricochet", SND_CHANNEL_ITEM, 0, true, NULL ) ) {
547 float len = velocity.Length();
548 if ( len > BOUNCE_SOUND_MIN_VELOCITY ) {
549 SetSoundVolume( len > BOUNCE_SOUND_MAX_VELOCITY ? 1.0f : idMath::Sqrt( len - BOUNCE_SOUND_MIN_VELOCITY ) * ( 1.0f / idMath::Sqrt( BOUNCE_SOUND_MAX_VELOCITY - BOUNCE_SOUND_MIN_VELOCITY ) ) );
550 StartSound( "snd_bounce", SND_CHANNEL_ANY, 0, true, NULL );
551 }
552 }
553 return false;
554 }
555 }
556
557 SetOrigin( collision.endpos );
558 SetAxis( collision.endAxis );
559
560 // unlink the clip model because we no longer need it
561 GetPhysics()->UnlinkClip();
562
563 damageDefName = spawnArgs.GetString( "def_damage" );
564
565 ignore = NULL;
566
567 // if the hit entity takes damage
568 if ( ent->fl.takedamage ) {
569 if ( damagePower ) {
570 damageScale = damagePower;
571 } else {
572 damageScale = 1.0f;
573 }
574
575 // if the projectile owner is a player
576 if ( owner.GetEntity() && owner.GetEntity()->IsType( idPlayer::Type ) ) {
577 // if the projectile hit an actor
578 if ( ent->IsType( idActor::Type ) ) {
579 idPlayer *player = static_cast<idPlayer *>( owner.GetEntity() );
580 player->AddProjectileHits( 1 );
581 damageScale *= player->PowerUpModifier( PROJECTILE_DAMAGE );
582 }
583 }
584
585 if ( damageDefName[0] != '\0' ) {
586 ent->Damage( this, owner.GetEntity(), dir, damageDefName, damageScale, CLIPMODEL_ID_TO_JOINT_HANDLE( collision.c.id ) );
587 ignore = ent;
588 }
589 }
590
591 // if the projectile causes a damage effect
592 if ( spawnArgs.GetBool( "impact_damage_effect" ) ) {
593 // if the hit entity has a special damage effect
594 if ( ent->spawnArgs.GetBool( "bleed" ) ) {
595 ent->AddDamageEffect( collision, velocity, damageDefName );
596 } else {
597 AddDefaultDamageEffect( collision, velocity );
598 }
599 }
600
601 Explode( collision, ignore );
602
603 return true;
604}

611void idProjectile::DefaultDamageEffect( idEntity *soundEnt, const idDict &projectileDef, const trace_t &collision, const idVec3 &velocity ) {
612 const char *decal, *sound, *typeName;
613 surfTypes_t materialType;
614
615 if ( collision.c.material != NULL ) {
616 materialType = collision.c.material->GetSurfaceType();
617 } else {
618 materialType = SURFTYPE_METAL;
619 }
620
621 // get material type name
622 typeName = gameLocal.sufaceTypeNames[ materialType ];
623
624 // play impact sound
625 sound = projectileDef.GetString( va( "snd_%s", typeName ) );
626 if ( *sound == '\0' ) {
627 sound = projectileDef.GetString( "snd_metal" );
628 }
629 if ( *sound == '\0' ) {
630 sound = projectileDef.GetString( "snd_impact" );
631 }
632 if ( *sound != '\0' ) {
633 soundEnt->StartSoundShader( declManager->FindSound( sound ), SND_CHANNEL_BODY, 0, false, NULL );
634 }
635
636 // project decal
637 decal = projectileDef.GetString( va( "mtr_detonate_%s", typeName ) );
638 if ( *decal == '\0' ) {
639 decal = projectileDef.GetString( "mtr_detonate" );
640 }
641 if ( *decal != '\0' ) {
642 gameLocal.ProjectDecal( collision.c.point, -collision.c.normal, 8.0f, true, projectileDef.GetFloat( "decal_size", "6.0" ), decal );
643 }
644}

702/*
703================
704idProjectile::Fizzle
705================
706*/
707void idProjectile::Fizzle( void ) {
708
709 if ( state == EXPLODED || state == FIZZLED ) {
710 return;
711 }
712
713 StopSound( SND_CHANNEL_BODY, false );
714 StartSound( "snd_fizzle", SND_CHANNEL_BODY, 0, false, NULL );
715
716 // fizzle FX
717 const char *psystem = spawnArgs.GetString( "smoke_fuse" );
718 if ( psystem && *psystem ) {
719//FIXME:SMOKE gameLocal.particles->SpawnParticles( GetPhysics()->GetOrigin(), vec3_origin, psystem );
720 }
721
722 // we need to work out how long the effects last and then remove them at that time
723 // for example, bullets have no real effects
724 if ( smokeFly && smokeFlyTime ) {
725 smokeFlyTime = 0;
726 }
727
728 fl.takedamage = false;
729 physicsObj.SetContents( 0 );
730 physicsObj.GetClipModel()->Unlink();
731 physicsObj.PutToRest();
732
733 Hide();
734 FreeLightDef();
735
736 state = FIZZLED;
737
738 if ( gameLocal.isClient ) {
739 return;

772void idProjectile::Explode( const trace_t &collision, idEntity *ignore ) {
773 const char *fxname, *light_shader, *sndExplode;
774 float light_fadetime;
775 idVec3 normal;
776 int removeTime;
777
778 if ( state == EXPLODED || state == FIZZLED ) {
779 return;
780 }
781
782 // stop sound
783 StopSound( SND_CHANNEL_BODY2, false );
784
785 // play explode sound
786 switch ( ( int ) damagePower ) {
787 case 2: sndExplode = "snd_explode2"; break;
788 case 3: sndExplode = "snd_explode3"; break;
789 case 4: sndExplode = "snd_explode4"; break;
790 default: sndExplode = "snd_explode"; break;
791 }
792 StartSound( sndExplode, SND_CHANNEL_BODY, 0, true, NULL );
793
794 // we need to work out how long the effects last and then remove them at that time
795 // for example, bullets have no real effects
796 if ( smokeFly && smokeFlyTime ) {
797 smokeFlyTime = 0;
798 }
799
800 Hide();
801 FreeLightDef();
802
803 if ( spawnArgs.GetVector( "detonation_axis", "", normal ) ) {
804 GetPhysics()->SetAxis( normal.ToMat3() );
805 }
806 GetPhysics()->SetOrigin( collision.endpos + 2.0f * collision.c.normal );
807
808 // default remove time
809 removeTime = spawnArgs.GetInt( "remove_time", "1500" );
810
811 // change the model, usually to a PRT
812 fxname = NULL;
813 if ( g_testParticle.GetInteger() == TEST_PARTICLE_IMPACT ) {
814 fxname = g_testParticleName.GetString();
815 } else {
816 fxname = spawnArgs.GetString( "model_detonate" );
817 }
818
819 int surfaceType = collision.c.material != NULL ? collision.c.material->GetSurfaceType() : SURFTYPE_METAL;
820 if ( !( fxname && *fxname ) ) {
821 if ( ( surfaceType == SURFTYPE_NONE ) || ( surfaceType == SURFTYPE_METAL ) || ( surfaceType == SURFTYPE_STONE ) ) {
822 fxname = spawnArgs.GetString( "model_smokespark" );
823 } else if ( surfaceType == SURFTYPE_RICOCHET ) {
824 fxname = spawnArgs.GetString( "model_ricochet" );
825 } else {
826 fxname = spawnArgs.GetString( "model_smoke" );
827 }
828 }
829
830 if ( fxname && *fxname ) {
831 SetModel( fxname );
832 renderEntity.shaderParms[SHADERPARM_RED] =
833 renderEntity.shaderParms[SHADERPARM_GREEN] =
834 renderEntity.shaderParms[SHADERPARM_BLUE] =
835 renderEntity.shaderParms[SHADERPARM_ALPHA] = 1.0f;
836 renderEntity.shaderParms[SHADERPARM_TIMEOFFSET] = -MS2SEC( gameLocal.time );
837 renderEntity.shaderParms[SHADERPARM_DIVERSITY] = gameLocal.random.CRandomFloat();
838 Show();
839 removeTime = ( removeTime > 3000 ) ? removeTime : 3000;
840 }
841
842 // explosion light
843 light_shader = spawnArgs.GetString( "mtr_explode_light_shader" );
844 if ( *light_shader ) {
845 renderLight.shader = declManager->FindMaterial( light_shader, false );
846 renderLight.pointLight = true;
847 renderLight.lightRadius[0] =
848 renderLight.lightRadius[1] =
849 renderLight.lightRadius[2] = spawnArgs.GetFloat( "explode_light_radius" );
850 spawnArgs.GetVector( "explode_light_color", "1 1 1", lightColor );
851 renderLight.shaderParms[SHADERPARM_RED] = lightColor.x;
852 renderLight.shaderParms[SHADERPARM_GREEN] = lightColor.y;
853 renderLight.shaderParms[SHADERPARM_BLUE] = lightColor.z;
854 renderLight.shaderParms[SHADERPARM_ALPHA] = 1.0f;
855 renderLight.shaderParms[SHADERPARM_TIMEOFFSET] = -MS2SEC( gameLocal.time );
856 light_fadetime = spawnArgs.GetFloat( "explode_light_fadetime", "0.5" );
857 lightStartTime = gameLocal.time;
858 lightEndTime = gameLocal.time + SEC2MS( light_fadetime );
859 BecomeActive( TH_THINK );
860 }
861
862 fl.takedamage = false;
863 physicsObj.SetContents( 0 );
864 physicsObj.PutToRest();
865
866 state = EXPLODED;
867
868 if ( gameLocal.isClient ) {
869 return;
870 }
871
872 // alert the ai
873 gameLocal.AlertAI( owner.GetEntity() );
874
875 // bind the projectile to the impact entity if necesary
876 if ( gameLocal.entities[collision.c.entityNum] && spawnArgs.GetBool( "bindOnImpact" ) ) {
877 Bind( gameLocal.entities[collision.c.entityNum], true );
878 }
879
880 // splash damage
881 if ( !projectileFlags.noSplashDamage ) {
882 float delay = spawnArgs.GetFloat( "delay_splash" );
883 if ( delay ) {
884 if ( removeTime < delay * 1000 ) {
885 removeTime = ( delay + 0.10 ) * 1000;
886 }
887 PostEventSec( &EV_RadiusDamage, delay, ignore );
888 } else {
889 Event_RadiusDamage( ignore );
890 }
891 }

894 int fxdebris = spawnArgs.GetInt( "debris_count" );
895 if ( fxdebris ) {

912 idDebris *debris = static_cast<idDebris *>(ent);
913 debris->Create( owner.GetEntity(), physicsObj.GetOrigin(), dir.ToMat3() );
914 debris->Launch();
915 }
916 }
917 debris = gameLocal.FindEntityDefDict( "projectile_shrapnel", false );
918 if ( debris ) {
919 int amount = gameLocal.random.RandomInt( fxdebris );
920 for ( int i = 0; i < amount; i++ ) {
921 idEntity *ent;
922 idVec3 dir;
923 dir.x = gameLocal.random.CRandomFloat() * 8.0f;
924 dir.y = gameLocal.random.CRandomFloat() * 8.0f;

941 PostEventMS( &EV_Remove, removeTime );

1023bool idProjectile::ClientPredictionCollide( idEntity *soundEnt, const idDict &projectileDef, const trace_t &collision, const idVec3 &velocity, bool addDamageEffect ) {
1024 idEntity *ent;
1025
1026 // remove projectile when a 'noimpact' surface is hit
1027 if ( collision.c.material && ( collision.c.material->GetSurfaceFlags() & SURF_NOIMPACT ) ) {

1029 }
1030
1031 // get the entity the projectile collided with
1032 ent = gameLocal.entities[ collision.c.entityNum ];
1033 if ( ent == NULL ) {
1034 return false;
1035 }
1036
1037 // don't do anything if hitting a noclip player
1038 if ( ent->IsType( idPlayer::Type ) && static_cast<idPlayer *>( ent )->noclip ) {
1039 return false;
1040 }
1041
1042 if ( ent->IsType( idActor::Type ) || ( ent->IsType( idAFAttachment::Type ) && static_cast<const idAFAttachment*>(ent)->GetBody()->IsType( idActor::Type ) ) ) {
1043 if ( !projectileDef.GetBool( "detonate_on_actor" ) ) {
1044 return false;
1045 }
1046 } else {
1047 if ( !projectileDef.GetBool( "detonate_on_world" ) ) {
1048 return false;
1049 }
1050 }
1051
1052 // if the projectile causes a damage effect
1053 if ( addDamageEffect && projectileDef.GetBool( "impact_damage_effect" ) ) {
1054 // if the hit entity does not have a special damage effect
1055 if ( !ent->spawnArgs.GetBool( "bleed" ) ) {
1056 // predict damage effect
1057 DefaultDamageEffect( soundEnt, projectileDef, collision, velocity );
1058 }
1059 }
1060 return true;
1061}

1102/*
1103================
1104idProjectile::ReadFromSnapshot
1105================
1106*/
1107void idProjectile::ReadFromSnapshot( const idBitMsgDelta &msg ) {
1108 projectileState_t newState;
1109
1110 owner.SetSpawnId( msg.ReadBits( 32 ) );
1111 newState = (projectileState_t) msg.ReadBits( 3 );
1112 if ( msg.ReadBits( 1 ) ) {
1113 Hide();
1114 } else {
1115 Show();
1116 }
1117
1118 while( state != newState ) {
1119 switch( state ) {
1120 case SPAWNED: {
1121 Create( owner.GetEntity(), vec3_origin, idVec3( 1, 0, 0 ) );
1122 break;
1123 }
1124 case CREATED: {
1125 // the right origin and direction are required if you want bullet traces
1126 Launch( vec3_origin, idVec3( 1, 0, 0 ), vec3_origin );
1127 break;
1128 }
1129 case LAUNCHED: {
1130 if ( newState == FIZZLED ) {
1131 Fizzle();
1132 } else {
1133 trace_t collision;
1134 memset( &collision, 0, sizeof( collision ) );
1135 collision.endAxis = GetPhysics()->GetAxis();
1136 collision.endpos = GetPhysics()->GetOrigin();
1137 collision.c.point = GetPhysics()->GetOrigin();
1138 collision.c.normal.Set( 0, 0, 1 );
1139 Explode( collision, NULL );
1140 }
1141 break;
1142 }
1143 case FIZZLED:
1144 case EXPLODED: {
1145 StopSound( SND_CHANNEL_BODY2, false );
1146 gameEdit->ParseSpawnArgsToRenderEntity( &spawnArgs, &renderEntity );
1147 state = SPAWNED;
1148 break;
1149 }
1150 }
1151 }
1152
1153 if ( msg.ReadBits( 1 ) ) {
1154 physicsObj.ReadFromSnapshot( msg );
1155 } else {
1156 idVec3 origin;
1157 idVec3 velocity;
1158 idVec3 tmp;
1159 idMat3 axis;
1160
1161 origin.x = msg.ReadFloat();
1162 origin.y = msg.ReadFloat();
1163 origin.z = msg.ReadFloat();
1164
1165 velocity.x = msg.ReadDeltaFloat( 0.0f, RB_VELOCITY_EXPONENT_BITS, RB_VELOCITY_MANTISSA_BITS );
1166 velocity.y = msg.ReadDeltaFloat( 0.0f, RB_VELOCITY_EXPONENT_BITS, RB_VELOCITY_MANTISSA_BITS );
1167 velocity.z = msg.ReadDeltaFloat( 0.0f, RB_VELOCITY_EXPONENT_BITS, RB_VELOCITY_MANTISSA_BITS );
1168
1169 physicsObj.SetOrigin( origin );
1170 physicsObj.SetLinearVelocity( velocity );
1171
1172 // align z-axis of model with the direction
1173 velocity.NormalizeFast();
1174 axis = velocity.ToMat3();
1175 tmp = axis[2];
1176 axis[2] = axis[0];
1177 axis[0] = -tmp;
1178 physicsObj.SetAxis( axis );
1179 }
1180
1181 if ( msg.HasChanged() ) {
1182 UpdateVisuals();
1183 }
1184}

1324/*
1325================
1326idGuidedProjectile::Think
1327================
1328*/
1329void idGuidedProjectile::Think( void ) {
1330 idVec3 dir;
1331 idVec3 seekPos;
1332 idVec3 velocity;
1333 idVec3 nose;
1334 idVec3 tmp;
1335 idMat3 axis;
1336 idAngles dirAng;
1337 idAngles diff;
1338 float dist;
1339 float frac;
1340 int i;
1341
1342 if ( state == LAUNCHED && !unGuided ) {
1343
1344 GetSeekPos( seekPos );
1345
1346 if ( rndUpdateTime < gameLocal.time ) {
1347 rndAng[ 0 ] = rndScale[ 0 ] * gameLocal.random.CRandomFloat();
1348 rndAng[ 1 ] = rndScale[ 1 ] * gameLocal.random.CRandomFloat();
1349 rndAng[ 2 ] = rndScale[ 2 ] * gameLocal.random.CRandomFloat();
1350 rndUpdateTime = gameLocal.time + 200;
1351 }
1352
1353 nose = physicsObj.GetOrigin() + 10.0f * physicsObj.GetAxis()[0];
1354
1355 dir = seekPos - nose;
1356 dist = dir.Normalize();
1357 dirAng = dir.ToAngles();
1358
1359 // make it more accurate as it gets closer
1360 frac = dist / clamp_dist;
1361 if ( frac > 1.0f ) {
1362 frac = 1.0f;
1363 }
1364
1365 diff = dirAng - angles + rndAng * frac;
1366
1367 // clamp the to the max turn rate
1368 diff.Normalize180();
1369 for( i = 0; i < 3; i++ ) {
1370 if ( diff[ i ] > turn_max ) {
1371 diff[ i ] = turn_max;
1372 } else if ( diff[ i ] < -turn_max ) {
1373 diff[ i ] = -turn_max;
1374 }
1375 }
1376 angles += diff;
1377
1378 // make the visual model always points the dir we're traveling
1379 dir = angles.ToForward();
1380 velocity = dir * speed;
1381
1382 if ( burstMode && dist < burstDist ) {
1383 unGuided = true;
1384 velocity *= burstVelocity;
1385 }
1386
1387 physicsObj.SetLinearVelocity( velocity );
1388
1389 // align z-axis of model with the direction
1390 axis = dir.ToMat3();
1391 tmp = axis[2];
1392 axis[2] = axis[0];
1393 axis[0] = -tmp;
1394
1395 GetPhysics()->SetAxis( axis );
1396 }
1397
1398 idProjectile::Think();
1399}

1406void idGuidedProjectile::Launch( const idVec3 &start, const idVec3 &dir, const idVec3 &pushVelocity, const float timeSinceFire, const float launchPower, float dmgPower ) {
1407 idProjectile::Launch( start, dir, pushVelocity, timeSinceFire, launchPower, dmgPower );
1408 if ( owner.GetEntity() ) {
1409 if ( owner.GetEntity()->IsType( idAI::Type ) ) {
1410 enemy = static_cast<idAI *>( owner.GetEntity() )->GetEnemy();
1411 } else if ( owner.GetEntity()->IsType( idPlayer::Type ) ) {
1412 trace_t tr;
1413 idPlayer *player = static_cast<idPlayer*>( owner.GetEntity() );
1414 idVec3 start = player->GetEyePosition();
1415 idVec3 end = start + player->viewAxis[0] * 1000.0f;
1416 gameLocal.clip.TracePoint( tr, start, end, MASK_SHOT_RENDERMODEL | CONTENTS_BODY, owner.GetEntity() );
1417 if ( tr.fraction < 1.0f ) {
1418 enemy = gameLocal.GetTraceEntity( tr );
1419 }
1420 // ignore actors on the player's team
1421 if ( enemy.GetEntity() == NULL || !enemy.GetEntity()->IsType( idActor::Type ) || ( static_cast<idActor *>( enemy.GetEntity() )->team == player->team ) ) {
1422 enemy = player->EnemyWithMostHealth();
1423 }
1424 }
1425 }
1426 const idVec3 &vel = physicsObj.GetLinearVelocity();
1427 angles = vel.ToAngles();
1428 speed = vel.Length();
1429 rndScale = spawnArgs.GetAngles( "random", "15 15 0" );
1430 turn_max = spawnArgs.GetFloat( "turn_max", "180" ) / ( float )USERCMD_HZ;
1431 clamp_dist = spawnArgs.GetFloat( "clamp_dist", "256" );
1432 burstMode = spawnArgs.GetBool( "burstMode" );

40class idAASFileLocal : public idAASFile {
41 friend class idAASBuild;
42 friend class idAASReach;
43 friend class idAASCluster;
44public:
45 idAASFileLocal( void );
46 virtual ~idAASFileLocal( void );
47
48public:
49 virtual idVec3 EdgeCenter( int edgeNum ) const;
50 virtual idVec3 FaceCenter( int faceNum ) const;
51 virtual idVec3 AreaCenter( int areaNum ) const;
52
53 virtual idBounds EdgeBounds( int edgeNum ) const;
54 virtual idBounds FaceBounds( int faceNum ) const;
55 virtual idBounds AreaBounds( int areaNum ) const;
56
57 virtual int PointAreaNum( const idVec3 &origin ) const;
58 virtual int PointReachableAreaNum( const idVec3 &origin, const idBounds &searchBounds, const int areaFlags, const int excludeTravelFlags ) const;
59 virtual int BoundsReachableAreaNum( const idBounds &bounds, const int areaFlags, const int excludeTravelFlags ) const;
60 virtual void PushPointIntoAreaNum( int areaNum, idVec3 &point ) const;
61 virtual bool Trace( aasTrace_t &trace, const idVec3 &start, const idVec3 &end ) const;
62 virtual void PrintInfo( void ) const;
63
64public:
65 bool Load( const idStr &fileName, unsigned int mapFileCRC );
66 bool Write( const idStr &fileName, unsigned int mapFileCRC );
67
68 int MemorySize( void ) const;
69 void ReportRoutingEfficiency( void ) const;
70 void Optimize( void );
71 void LinkReversedReachability( void );
72 void FinishAreas( void );
73
74 void Clear( void );
75 void DeleteReachabilities( void );
76 void DeleteClusters( void );
77
78private:
79 bool ParseIndex( idLexer &src, idList<aasIndex_t> &indexes );
80 bool ParsePlanes( idLexer &src );
81 bool ParseVertices( idLexer &src );
82 bool ParseEdges( idLexer &src );
83 bool ParseFaces( idLexer &src );
84 bool ParseReachabilities( idLexer &src, int areaNum );
85 bool ParseAreas( idLexer &src );
86 bool ParseNodes( idLexer &src );
87 bool ParsePortals( idLexer &src );
88 bool ParseClusters( idLexer &src );
89
90private:
91 int BoundsReachableAreaNum_r( int nodeNum, const idBounds &bounds, const int areaFlags, const int excludeTravelFlags ) const;
92 void MaxTreeDepth_r( int nodeNum, int &depth, int &maxDepth ) const;
93 int MaxTreeDepth( void ) const;
94 int AreaContentsTravelFlags( int areaNum ) const;
95 idVec3 AreaReachableGoal( int areaNum ) const;
96 int NumReachabilities( void ) const;
97};

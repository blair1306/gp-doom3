39#define dabs(a) (((a)<0) ? -(a) : (a))
40#define CLAMP(v,l,h) ((v)<(l) ? (l) : (v)>(h) ? (h) : v)
41#define xswap(a,b) { a^=b; b^=a; a^=b; }
42#define lum(a) ( 0.2990*(a>>16) + 0.5870*((a>>8)&0xff) + 0.1140*(a&0xff) )
43#define gsign(a) ((a) < 0 ? -1 : 1)
44#define mnint(a) ((a) < 0 ? (int)(a - 0.5) : (int)(a + 0.5))
45#define mmax(a, b) ((a) > (b) ? (a) : (b))
46#define mmin(a, b) ((a) < (b) ? (a) : (b))
47#define RGBDIST( src0, src1 ) ( ((src0[0]-src1[0])*(src0[0]-src1[0])) + \
48 ((src0[1]-src1[1])*(src0[1]-src1[1])) + \
49 ((src0[2]-src1[2])*(src0[2]-src1[2])) )
50
51#define RGBADIST( src0, src1 ) ( ((src0[0]-src1[0])*(src0[0]-src1[0])) + \
52 ((src0[1]-src1[1])*(src0[1]-src1[1])) + \
53 ((src0[2]-src1[2])*(src0[2]-src1[2])) + \
54 ((src0[3]-src1[3])*(src0[3]-src1[3])) )

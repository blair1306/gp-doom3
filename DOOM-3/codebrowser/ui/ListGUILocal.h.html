29#ifndef __LISTGUILOCAL_H__
30#define __LISTGUILOCAL_H__
31
32/*
33===============================================================================
34
35 feed data to a listDef
36 each item has an id and a display string
37
38===============================================================================
39*/
40
41class idListGUILocal : protected idList<idStr>, public idListGUI {
42public:
43 idListGUILocal() { m_pGUI = NULL; m_water = 0; m_stateUpdates = true; }
44
45 // idListGUI interface
46 void Config( idUserInterface *pGUI, const char *name ) { m_pGUI = pGUI; m_name = name; }
47 void Add( int id, const idStr& s );
48 // use the element count as index for the ids
49 void Push( const idStr& s );
50 bool Del( int id );
51 void Clear( void );
52 int Num( void ) { return idList<idStr>::Num(); }
53 int GetSelection( char *s, int size, int sel = 0 ) const; // returns the id, not the list index (or -1)
54 void SetSelection( int sel );
55 int GetNumSelections();
56 bool IsConfigured( void ) const;
57 void SetStateChanges( bool enable );
58 void Shutdown( void );
59
60private:
61 idUserInterface * m_pGUI;
62 idStr m_name;
63 int m_water;
64 idList<int> m_ids;
65 bool m_stateUpdates;
66
67 void StateChanged();
68};
69
70#endif /* !__LISTGUILOCAL_H__ */

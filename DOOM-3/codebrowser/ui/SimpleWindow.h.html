69 idRectangle drawRect; // overall rect
70 idRectangle clientRect; // client area
71 idRectangle textRect;

73 int fontNum;

80 int textShadow;

82 idWinStr text;

85 idWinVec4 backColor;

96 idWindow * mParent;

98 idWinBool hideCursor;

29#ifndef __LISTGUI_H__
30#define __LISTGUI_H__
31
32/*
33===============================================================================
34
35 feed data to a listDef
36 each item has an id and a display string
37
38===============================================================================
39*/
40
41class idListGUI {
42public:
43 virtual ~idListGUI() { }
44
45 virtual void Config( idUserInterface *pGUI, const char *name ) = 0;
46 virtual void Add( int id, const idStr& s ) = 0;
47 // use the element count as index for the ids
48 virtual void Push( const idStr& s ) = 0;
49 virtual bool Del( int id ) = 0;
50 virtual void Clear( void ) = 0;
51 virtual int Num( void ) = 0;
52 virtual int GetSelection( char *s, int size, int sel = 0 ) const = 0; // returns the id, not the list index (or -1)
53 virtual void SetSelection( int sel ) = 0;
54 virtual int GetNumSelections() = 0;
55 virtual bool IsConfigured( void ) const = 0;
56 // by default, any modification to the list will trigger a full GUI refresh immediately
57 virtual void SetStateChanges( bool enable ) = 0;
58 virtual void Shutdown( void ) = 0;
59};
60
61#endif /* !__LISTGUI_H__ */

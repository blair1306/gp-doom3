29#ifndef __SLIDERWINDOW_H__
30#define __SLIDERWINDOW_H__
31
32class idUserInterfaceLocal;
33
34class idSliderWindow : public idWindow {
35public:
36 idSliderWindow(idUserInterfaceLocal *gui);
37 idSliderWindow(idDeviceContext *d, idUserInterfaceLocal *gui);
38 virtual ~idSliderWindow();
39
40 void InitWithDefaults(const char *_name, const idRectangle &rect, const idVec4 &foreColor, const idVec4 &matColor, const char *_background, const char *thumbShader, bool _vertical, bool _scrollbar);
41
42 void SetRange(float _low, float _high, float _step);
43 float GetLow() { return low; }
44 float GetHigh() { return high; }
45
46 void SetValue(float _value);
47 float GetValue() { return value; };
48
49 virtual size_t Allocated(){return idWindow::Allocated();};
50 virtual idWinVar * GetWinVarByName(const char *_name, bool winLookup = false, drawWin_t** owner = NULL);
51 virtual const char *HandleEvent(const sysEvent_t *event, bool *updateVisuals);
52 virtual void PostParse();
53 virtual void Draw(int time, float x, float y);
54 virtual void DrawBackground(const idRectangle &drawRect);
55 virtual const char *RouteMouseCoords(float xd, float yd);
56 virtual void Activate(bool activate, idStr &act);
57 virtual void SetBuddy(idWindow *buddy);
58
59 void RunNamedEvent( const char* eventName );
60
61private:
62 virtual bool ParseInternalVar(const char *name, idParser *src);
63 void CommonInit();
64 void InitCvar();
65 // true: read the updated cvar from cvar system
66 // false: write to the cvar system
67 // force == true overrides liveUpdate 0
68 void UpdateCvar( bool read, bool force = false );
69
70 idWinFloat value;
71 float low;
72 float high;
73 float thumbWidth;
74 float thumbHeight;
75 float stepSize;
76 float lastValue;
77 idRectangle thumbRect;
78 const idMaterial * thumbMat;
79 bool vertical;
80 bool verticalFlip;
81 bool scrollbar;
82 idWindow * buddyWin;
83 idStr thumbShader;
84
85 idWinStr cvarStr;
86 idCVar * cvar;
87 bool cvar_init;
88 idWinBool liveUpdate;
89 idWinStr cvarGroup;
90};
91
92#endif /* !__SLIDERWINDOW_H__ */

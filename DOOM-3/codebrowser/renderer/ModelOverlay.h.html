29#ifndef __MODELOVERLAY_H__
30#define __MODELOVERLAY_H__
31
32/*
33===============================================================================
34
35 Render model overlay for adding decals on top of dynamic models.
36
37===============================================================================
38*/
39
40const int MAX_OVERLAY_SURFACES = 16;
41
42typedef struct overlayVertex_s {
43 int vertexNum;
44 float st[2];
45} overlayVertex_t;
46
47typedef struct overlaySurface_s {
48 int surfaceNum;
49 int surfaceId;
50 int numIndexes;
51 glIndex_t * indexes;
52 int numVerts;
53 overlayVertex_t * verts;
54} overlaySurface_t;
55
56typedef struct overlayMaterial_s {
57 const idMaterial * material;
58 idList<overlaySurface_t *> surfaces;
59} overlayMaterial_t;
60
61
62class idRenderModelOverlay {
63public:
64 idRenderModelOverlay();
65 ~idRenderModelOverlay();
66
67 static idRenderModelOverlay *Alloc( void );
68 static void Free( idRenderModelOverlay *overlay );
69
70 // Projects an overlay onto deformable geometry and can be added to
71 // a render entity to allow decals on top of dynamic models.
72 // This does not generate tangent vectors, so it can't be used with
73 // light interaction shaders. Materials for overlays should always
74 // be clamped, because the projected texcoords can run well off the
75 // texture since no new clip vertexes are generated.
76 void CreateOverlay( const idRenderModel *model, const idPlane localTextureAxis[2], const idMaterial *material );
77
78 // Creates new model surfaces for baseModel, which should be a static instantiation of a dynamic model.
79 void AddOverlaySurfacesToModel( idRenderModel *baseModel );
80
81 // Removes overlay surfaces from the model.
82 static void RemoveOverlaySurfacesFromModel( idRenderModel *baseModel );
83
84 void ReadFromDemoFile( class idDemoFile *f );
85 void WriteToDemoFile( class idDemoFile *f ) const;
86
87private:
88 idList<overlayMaterial_t *> materials;
89
90 void FreeSurface( overlaySurface_t *surface );
91};
92
93#endif /* !__MODELOVERLAY_H__ */
